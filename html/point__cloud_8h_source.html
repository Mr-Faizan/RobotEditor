223// Functor for computing a hash from data stored within a point cloud.
224// Note that this can be quite slow. Two point clouds will have the same hash
225// only when all points have the same order and when all attribute values are
226// exactly the same.
227struct PointCloudHasher {
228 size_t operator()(const PointCloud &pc) const {
229 size_t hash = pc.num_points_;
230 hash = HashCombine(pc.attributes_.size(), hash);
231 for (int i = 0; i < GeometryAttribute::NAMED_ATTRIBUTES_COUNT; ++i) {
232 hash = HashCombine(pc.named_attribute_index_[i].size(), hash);
233 for (int j = 0; j < static_cast<int>(pc.named_attribute_index_[i].size());
234 ++j) {
235 hash = HashCombine(pc.named_attribute_index_[i][j], hash);
236 }
237 }
238 // Hash attributes.
239 for (int i = 0; i < static_cast<int>(pc.attributes_.size()); ++i) {
240 PointAttributeHasher att_hasher;
241 hash = HashCombine(att_hasher(*pc.attributes_[i]), hash);
242 }
243 // Hash metadata.
244 GeometryMetadataHasher metadata_hasher;
245 if (pc.metadata_) {
246 hash = HashCombine(metadata_hasher(*pc.metadata_), hash);
247 }
248 return hash;
249 }
250};
