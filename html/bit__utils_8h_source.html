32// Returns the number of '1' bits within the input 32 bit integer.
33inline int CountOneBits32(uint32_t n) {
34 n -= ((n >> 1) & 0x55555555);
35 n = ((n >> 2) & 0x33333333) + (n & 0x33333333);
36 return (((n + (n >> 4)) & 0xF0F0F0F) * 0x1010101) >> 24;
37}
38
39inline uint32_t ReverseBits32(uint32_t n) {
40 n = ((n >> 1) & 0x55555555) | ((n & 0x55555555) << 1);
41 n = ((n >> 2) & 0x33333333) | ((n & 0x33333333) << 2);
42 n = ((n >> 4) & 0x0F0F0F0F) | ((n & 0x0F0F0F0F) << 4);
43 n = ((n >> 8) & 0x00FF00FF) | ((n & 0x00FF00FF) << 8);
44 return (n >> 16) | (n << 16);
45}

47// Copies the |nbits| from the src integer into the |dst| integer using the
48// provided bit offsets |dst_offset| and |src_offset|.
49inline void CopyBits32(uint32_t *dst, int dst_offset, uint32_t src,
50 int src_offset, int nbits) {
51 const uint32_t mask = (~static_cast<uint32_t>(0)) >> (32 - nbits)
52 << dst_offset;
53 *dst = (*dst & (~mask)) | (((src >> src_offset) << dst_offset) & mask);
54}

56// Returns the location of the most significant bit in the input integer |n|.
57// The functionality is not defined for |n == 0|.
58inline int MostSignificantBit(uint32_t n) {
59#if defined(__GNUC__)
60 return 31 ^ __builtin_clz(n);
61#elif defined(_MSC_VER)
62
63 unsigned long where;
64 _BitScanReverse(&where, n);
65 return (int)where;
66#else
67 // TODO(fgalligan): Optimize this code.
68 int msb = -1;
69 while (n != 0) {
70 msb++;
71 n >>= 1;
72 }
73 return msb;
74#endif
75}

87// Helper function that converts a single signed integer value into an unsigned
88// integer symbol that can be encoded using an entropy encoder.
89template <class IntTypeT>
90typename std::make_unsigned<IntTypeT>::type ConvertSignedIntToSymbol(
91 IntTypeT val) {
92 typedef typename std::make_unsigned<IntTypeT>::type UnsignedType;
93 static_assert(std::is_integral<IntTypeT>::value, "IntTypeT is not integral.");
94 // Early exit if val is positive.
95 if (val >= 0) {
96 return static_cast<UnsignedType>(val) << 1;
97 }
98 val = -(val + 1); // Map -1 to 0, -2 to -1, etc..
99 UnsignedType ret = static_cast<UnsignedType>(val);
100 ret <<= 1;
101 ret |= 1;
102 return ret;
103}
104
105// Converts a single unsigned integer symbol encoded with an entropy encoder
106// back to a signed value.
107template <class IntTypeT>
108typename std::make_signed<IntTypeT>::type ConvertSymbolToSignedInt(
109 IntTypeT val) {
110 static_assert(std::is_integral<IntTypeT>::value, "IntTypeT is not integral.");
111 typedef typename std::make_signed<IntTypeT>::type SignedType;
112 const bool is_positive = !static_cast<bool>(val & 1);
113 val >>= 1;
114 if (is_positive) {
115 return static_cast<SignedType>(val);
116 }
117 SignedType ret = static_cast<SignedType>(val);
118 ret = -ret - 1;
119 return ret;
120}
