152struct MinMaxChooser;
153
154template <>
155struct MinMaxChooser<float> {
156 void operator()(float &min, float &max) {
157 max = -1e10f;
158 min = 1e10f;
159 }
160};
161template <>
162struct MinMaxChooser<double> {
163 void operator()(double &min, double &max) {
164 max = -1e10;
165 min = 1e10;
166 }
167};
168template <>
169struct MinMaxChooser<unsigned int> {
170 void operator()(unsigned int &min, unsigned int &max) {
171 max = 0;
172 min = (1u << (sizeof(unsigned int) * 8 - 1));
173 }
174};
175
176template <typename T>
177struct MinMaxChooser<aiVector3t<T>> {
178 void operator()(aiVector3t<T> &min, aiVector3t<T> &max) {
179 max = aiVector3t<T>(-1e10f, -1e10f, -1e10f);
180 min = aiVector3t<T>(1e10f, 1e10f, 1e10f);
181 }
182};
183template <typename T>
184struct MinMaxChooser<aiVector2t<T>> {

186 max = aiVector2t<T>(-1e10f, -1e10f);
187 min = aiVector2t<T>(1e10f, 1e10f);
188 }
189};

236template <typename T>
237inline void ArrayBounds(const T *in, unsigned int size, T &min, T &max) {
238 MinMaxChooser<T>()(min, max);
239 for (unsigned int i = 0; i < size; ++i) {
240 min = std::min(in[i], min);
241 max = std::max(in[i], max);
242 }
243}

334// -------------------------------------------------------------------------------
335// Utility post-process step to share the spatial sort tree between
336// all steps which use it to speedup its computations.
337class ComputeSpatialSortProcess : public BaseProcess {
338 bool IsActive(unsigned int pFlags) const {
339 return nullptr != shared && 0 != (pFlags & (aiProcess_CalcTangentSpace |
340 aiProcess_GenNormals | aiProcess_JoinIdenticalVertices));
341 }
342
343 void Execute(aiScene *pScene) {
344 typedef std::pair<SpatialSort, ai_real> _Type;
345 ASSIMP_LOG_DEBUG("Generate spatially-sorted vertex cache");
346
347 std::vector<_Type> *p = new std::vector<_Type>(pScene->mNumMeshes);
348 std::vector<_Type>::iterator it = p->begin();
349
350 for (unsigned int i = 0; i < pScene->mNumMeshes; ++i, ++it) {
351 aiMesh *mesh = pScene->mMeshes[i];
352 _Type &blubb = *it;
353 blubb.first.Fill(mesh->mVertices, mesh->mNumVertices, sizeof(aiVector3D));
354 blubb.second = ComputePositionEpsilon(mesh);
355 }
356
357 shared->AddProperty(AI_SPP_SPATIAL_SORT, p);
358 }
359};
360
361// -------------------------------------------------------------------------------
362// ... and the same again to cleanup the whole stuff
363class DestroySpatialSortProcess : public BaseProcess {
364 bool IsActive(unsigned int pFlags) const {
365 return nullptr != shared && 0 != (pFlags & (aiProcess_CalcTangentSpace |
366 aiProcess_GenNormals | aiProcess_JoinIdenticalVertices));
367 }
368
369 void Execute(aiScene * /*pScene*/) {
370 shared->RemoveProperty(AI_SPP_SPATIAL_SORT);
371 }
372};
