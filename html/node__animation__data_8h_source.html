28// This class is used to store information and data for animations that only
29// affect the nodes.
30// TODO(fgalligan): Think about changing the name of this class now that Skin
31// is using it.
32class NodeAnimationData {
33 public:
34 enum class Type { SCALAR, VEC3, VEC4, MAT4 };
35
36 NodeAnimationData() : type_(Type::SCALAR), count_(0), normalized_(false) {}
37
38 void Copy(const NodeAnimationData &src) {
39 type_ = src.type_;
40 count_ = src.count_;
41 normalized_ = src.normalized_;
42 data_ = src.data_;
43 }
44
45 Type type() const { return type_; }
46 int count() const { return count_; }
47 bool normalized() const { return normalized_; }
48
49 std::vector<float> *GetMutableData() { return &data_; }
50 const std::vector<float> *GetData() const { return &data_; }
51
52 void SetType(Type type) { type_ = type; }
53 void SetCount(int count) { count_ = count; }
54 void SetNormalized(bool normalized) { normalized_ = normalized; }
55
56 int ComponentSize() const { return sizeof(float); }
57 int NumComponents() const {
58 switch (type_) {
59 case Type::SCALAR:
60 return 1;
61 case Type::VEC3:
62 return 3;
63 case Type::MAT4:
64 return 16;
65 default:
66 return 4;
67 }
68 }
69
70 std::string TypeAsString() const {
71 switch (type_) {
72 case Type::SCALAR:
73 return "SCALAR";
74 case Type::VEC3:
75 return "VEC3";
76 case Type::MAT4:
77 return "MAT4";
78 default:
79 return "VEC4";
80 }
81 }
82
83 bool operator==(const NodeAnimationData &nad) const {
84 return type_ == nad.type_ && count_ == nad.count_ &&
85 normalized_ == nad.normalized_ && data_ == nad.data_;
86 }
87
88 private:
89 Type type_;
90 int count_;
91 bool normalized_;
92 std::vector<float> data_;
93};

95// Wrapper class for hashing NodeAnimationData. When using different containers,
96// this class is preferable instead of copying the data in NodeAnimationData
97// every time.
98class NodeAnimationDataHash {
99 public:
100 NodeAnimationDataHash() = delete;
101 NodeAnimationDataHash &operator=(const NodeAnimationDataHash &) = delete;
102 NodeAnimationDataHash(NodeAnimationDataHash &&) = delete;
103 NodeAnimationDataHash &operator=(NodeAnimationDataHash &&) = delete;
104
105 explicit NodeAnimationDataHash(const NodeAnimationData *nad)
106 : node_animation_data_(nad) {
107 hash_ = NodeAnimationDataHash::HashNodeAnimationData(*node_animation_data_);
108 }
109
110 NodeAnimationDataHash(const NodeAnimationDataHash &nadh) {
111 node_animation_data_ = nadh.node_animation_data_;
112 hash_ = nadh.hash_;
113 }
114
115 bool operator==(const NodeAnimationDataHash &nadh) const {
116 return *node_animation_data_ == *nadh.node_animation_data_;
117 }
118
119 struct Hash {
120 size_t operator()(const NodeAnimationDataHash &nadh) const {
121 return nadh.hash_;
122 }
123 };
124
125 const NodeAnimationData *GetNodeAnimationData() {
126 return node_animation_data_;
127 }
128
129 private:
130 // Returns a hash of |nad|.
131 static size_t HashNodeAnimationData(const NodeAnimationData &nad) {
132 size_t hash = 79; // Magic number.
133 hash = HashCombine(static_cast<int>(nad.type()), hash);
134 hash = HashCombine(nad.count(), hash);
135 hash = HashCombine(nad.normalized(), hash);
136 const uint64_t data_hash =
137 FingerprintString(reinterpret_cast<const char *>(nad.GetData()->data()),
138 nad.GetData()->size() * sizeof(float));
139 hash = HashCombine(data_hash, hash);
140 return hash;
141 }
142
143 const NodeAnimationData *node_animation_data_;
144 size_t hash_;
145};
