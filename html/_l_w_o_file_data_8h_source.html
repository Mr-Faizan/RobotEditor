347struct VMapEntry {
348 explicit VMapEntry(unsigned int _dims) :
349 dims(_dims) {}
350
351 virtual ~VMapEntry() = default;
352
354 virtual void Allocate(unsigned int num) {
355 if (!rawData.empty())
356 return; // return if already allocated
357
358 const unsigned int m = num * dims;
359 rawData.reserve(m + (m >> 2u)); // 25% as extra storage for VMADs
360 rawData.resize(m, 0.f);
361 abAssigned.resize(num, false);
362 }
363
364 std::string name;
365 unsigned int dims;
366
367 std::vector<float> rawData;
368 std::vector<bool> abAssigned;
369};

374struct VColorChannel : public VMapEntry {
375 VColorChannel() :
376 VMapEntry(4) {}
377
380 virtual void Allocate(unsigned int num) {
381 if (!rawData.empty())
382 return; // return if already allocated
383
384 unsigned int m = num * dims;
385 rawData.reserve(m + (m >> 2u)); // 25% as extra storage for VMADs
386 rawData.resize(m);
387
388 for (aiColor4D *p = (aiColor4D *)&rawData[0]; p < (aiColor4D *)&rawData[m - 1]; ++p)
389 p->a = 1.f;
390
391 abAssigned.resize(num, false);
392 }
393};
