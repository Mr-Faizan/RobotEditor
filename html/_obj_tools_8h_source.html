112template <class char_t>
113inline char_t skipLine(char_t it, char_t end, unsigned int &uiLine) {
114 if (it >= end) {
115 return it;
116 }
117
118 while (!isEndOfBuffer(it, end) && !IsLineEnd(*it)) {
119 ++it;
120 }
121
122 if (it != end) {
123 ++it;
124 ++uiLine;
125 }
126 // fix .. from time to time there are spaces at the beginning of a material line
127 while (it != end && (*it == '\t' || *it == ' ')) {
128 ++it;
129 }
130
131 return it;
132}

142template <class char_t>
143inline char_t getName(char_t it, char_t end, std::string &name) {
144 name = "";
145 if (isEndOfBuffer(it, end)) {
146 return end;
147 }
148
149 char *pStart = &(*it);
150 while (!isEndOfBuffer(it, end) && !IsLineEnd(*it)) {
151 ++it;
152 }
153
154 while (IsSpace(*it)) {
155 --it;
156 }
157 // Get name
158 // if there is no name, and the previous char is a separator, come back to start
159 while (&(*it) < pStart) {
160 ++it;
161 }
162 std::string strName(pStart, &(*it));
163 if (!strName.empty()) {
164 name = strName;
165 }
166
167
168 return it;
169}

179template <class char_t>
180inline char_t getNameNoSpace(char_t it, char_t end, std::string &name) {
181 name = "";
182 if (isEndOfBuffer(it, end)) {
183 return end;
184 }
185
186 char *pStart = &(*it);
187 while (!isEndOfBuffer(it, end) && !IsLineEnd(*it) && !IsSpaceOrNewLine(*it)) {
188 ++it;
189 }
190
191 while (isEndOfBuffer(it, end) || IsLineEnd(*it) || IsSpaceOrNewLine(*it)) {
192 --it;
193 }
194 ++it;
195
196 // Get name
197 // if there is no name, and the previous char is a separator, come back to start
198 while (&(*it) < pStart) {
199 ++it;
200 }
201 std::string strName(pStart, &(*it));
202 if (!strName.empty()) {
203 name = strName;
204 }
205
206 return it;

217template <class char_t>
218inline char_t CopyNextWord(char_t it, char_t end, char *pBuffer, size_t length) {
219 size_t index = 0;
220 it = getNextWord<char_t>(it, end);
221 while (!IsSpaceOrNewLine(*it) && !isEndOfBuffer(it, end)) {
222 pBuffer[index] = *it;
223 ++index;
224 if (index == length - 1) {
225 break;
226 }
227 ++it;
228 }
229 pBuffer[index] = '\0';
230 return it;
231}

256template <class T>
257bool hasLineEnd(T it, T end) {
258 bool hasLineEnd = false;
259 while (!isEndOfBuffer(it, end)) {
260 ++it;
261 if (IsLineEnd(it)) {
262 hasLineEnd = true;
263 break;
264 }
265 }
266
