336// Finds the item in the given library by its reference, throws if not found
337template <typename Type>
338const Type &ColladaParser::ResolveLibraryReference(const std::map<std::string, Type> &pLibrary, const std::string &pURL) const {
339 typename std::map<std::string, Type>::const_iterator it = pLibrary.find(pURL);
340 if (it == pLibrary.end()) {
341 throw DeadlyImportError("Unable to resolve library reference \"", pURL, "\".");
342 }
343 return it->second;
344}
