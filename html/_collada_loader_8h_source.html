59struct ColladaMeshIndex {
60 std::string mMeshID;
61 size_t mSubMesh;
62 std::string mMaterial;
63 ColladaMeshIndex(const std::string &pMeshID, size_t pSubMesh, const std::string &pMaterial) :
64 mMeshID(pMeshID), mSubMesh(pSubMesh), mMaterial(pMaterial) {
65 ai_assert(!pMeshID.empty());
66 }
67
68 bool operator<(const ColladaMeshIndex &p) const {
69 if (mMeshID == p.mMeshID) {
70 if (mSubMesh == p.mSubMesh)
71 return mMaterial < p.mMaterial;
72 else
73 return mSubMesh < p.mSubMesh;
74 } else {
75 return mMeshID < p.mMeshID;
76 }
77 }
78};
