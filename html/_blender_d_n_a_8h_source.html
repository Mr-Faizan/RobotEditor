371 template <int error_policy>
372 struct _defaultInitializer {
373
374 template <typename T, unsigned int N>
375 void operator()(T (&out)[N], const char * = nullptr) {
376 for (unsigned int i = 0; i < N; ++i) {
377 out[i] = T();
378 }
379 }
380
381 template <typename T, unsigned int N, unsigned int M>
382 void operator()(T (&out)[N][M], const char * = nullptr) {
383 for (unsigned int i = 0; i < N; ++i) {
384 for (unsigned int j = 0; j < M; ++j) {
385 out[i][j] = T();
386 }
387 }
388 }
389
390 template <typename T>
391 void operator()(T &out, const char * = nullptr) {
392 out = T();
393 }
394 };

553struct FileBlockHead {
554 // points right after the header of the file block
555 StreamReaderAny::pos start;
556
557 std::string id;
558 size_t size;
559
560 // original memory address of the data
561 Pointer address;
562
563 // index into DNA
564 unsigned int dna_index;
565
566 // number of structure instances to follow
567 size_t num;
568
569 // file blocks are sorted by address to quickly locate specific memory addresses
570 bool operator<(const FileBlockHead &o) const {
571 return address.val < o.address.val;
572 }
573
574 // for std::upper_bound
575 operator const Pointer &() const {
576 return address;
577 }
578};
579
580// for std::upper_bound
581inline bool operator<(const Pointer &a, const Pointer &b) {
582 return a.val < b.val;
583}
