448template <typename TYPE>
449inline TYPE PLY::PropertyInstance::ConvertTo(
450 PLY::PropertyInstance::ValueUnion v, PLY::EDataType eType)
451{
452 switch (eType)
453 {
454 case EDT_Float:
455 return (TYPE)v.fFloat;
456 case EDT_Double:
457 return (TYPE)v.fDouble;
458
459 case EDT_UInt:
460 case EDT_UShort:
461 case EDT_UChar:
462 return (TYPE)v.iUInt;
463
464 case EDT_Int:
465 case EDT_Short:
466 case EDT_Char:
467 return (TYPE)v.iInt;
468 default: ;
469 };
470 return (TYPE)0;
471}
