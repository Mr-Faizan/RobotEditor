37// List of different variants of mesh attributes.
38enum MeshAttributeElementType {
39 // All corners attached to a vertex share the same attribute value. A typical
40 // example are the vertex positions and often vertex colors.
41 MESH_VERTEX_ATTRIBUTE = 0,
42 // The most general attribute where every corner of the mesh can have a
43 // different attribute value. Often used for texture coordinates or normals.
44 MESH_CORNER_ATTRIBUTE,
45 // All corners of a single face share the same value.
46 MESH_FACE_ATTRIBUTE
47};

313// Functor for computing a hash from data stored within a mesh.
314// Note that this can be quite slow. Two meshes will have the same hash only
315// when they have exactly the same connectivity and attribute values.
316struct MeshHasher {
317 size_t operator()(const Mesh &mesh) const {
318 PointCloudHasher pc_hasher;
319 size_t hash = pc_hasher(mesh);
320 // Hash faces.
321 for (FaceIndex i(0); i < static_cast<uint32_t>(mesh.faces_.size()); ++i) {
322 for (int j = 0; j < 3; ++j) {
323 hash = HashCombine(mesh.faces_[i][j].value(), hash);
324 }
325 }
326 return hash;
327 }
328};
