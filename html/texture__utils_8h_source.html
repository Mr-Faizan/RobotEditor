30// Helper class implementing various utilities operating on draco::Texture.
31class TextureUtils {
32 public:
33 // Returns |texture| image stem (file basename without extension) based on the
34 // source image filename or an empty string when source image is not set.
35 static std::string GetTargetStem(const Texture &texture);
36
37 // Returns |texture| image stem (file basename without extension) based on the
38 // source image filename or a name generated from |index| and |suffix| like
39 // "Texture5_BaseColor" when source image is not set.
40 static std::string GetOrGenerateTargetStem(const Texture &texture, int index,
41 const std::string &suffix);
42
43 // Returns |texture| format based on compression settings, the source image
44 // mime type or the source image filename.
45 static ImageFormat GetTargetFormat(const Texture &texture);
46
47 // Returns |texture| image file extension based on compression settings, the
48 // source image mime type or the source image filename.
49 static std::string GetTargetExtension(const Texture &texture);
50
51 // Returns |texture| format based on source image mime type or the source
52 // image filename.
53 static ImageFormat GetSourceFormat(const Texture &texture);
54
55 // Returns image format corresponding to a given image file |extension|. NONE
56 // is returned when |extension| is empty or unknown.
57 static ImageFormat GetFormat(const std::string &extension);
58
59 // Returns image file extension corresponding to a given image |format|. Empty
60 // extension is returned when the |format| is NONE.
61 static std::string GetExtension(ImageFormat format);
62
63 // Returns the number of channels required for encoding a |texture| from a
64 // given |material_library|, taking into account texture opacity and assuming
65 // that occlusion and metallic-roughness texture maps may share a texture.
66 // TODO(vytyaz): Move this and FindTextures() to MaterialLibrary class.
67 static int ComputeRequiredNumChannels(
68 const Texture &texture, const MaterialLibrary &material_library);
69
70 static std::vector<const Texture *> FindTextures(
71 const TextureMap::Type texture_type,
72 const MaterialLibrary *material_library);
73};
