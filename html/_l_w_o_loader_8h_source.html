459inline int LWOImporter::ReadVSizedIntLWO2(uint8_t *&inout) {
460 int i;
461 int c = *inout;
462 inout++;
463 if (c != 0xFF) {
464 i = c << 8;
465 c = *inout;
466 inout++;
467 i |= c;
468 } else {
469 c = *inout;
470 inout++;
471 i = c << 16;
472 c = *inout;
473 inout++;
474 i |= c << 8;
475 c = *inout;
476 inout++;
477 i |= c;
478 }
479 return i;
480}

483inline void LWOImporter::GetS0(std::string &out, unsigned int max) {
484 unsigned int iCursor = 0;
485 const char *sz = (const char *)mFileBuffer;
486 while (*mFileBuffer) {
487 if (++iCursor > max) {
488 ASSIMP_LOG_WARN("LWO: Invalid file, string is is too long");
489 break;
490 }
491 ++mFileBuffer;
492 }
493 size_t len = (size_t)((const char *)mFileBuffer - sz);
494 out = std::string(sz, len);
495 mFileBuffer += (len & 0x1 ? 1 : 2);
496}
