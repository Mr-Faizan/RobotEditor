33// The class provides access to a specific attribute which is stored in a
34// DataBuffer, such as normals or coordinates. However, the GeometryAttribute
35// class does not own the buffer and the buffer itself may store other data
36// unrelated to this attribute (such as data for other attributes in which case
37// we can have multiple GeometryAttributes accessing one buffer). Typically,
38// all attributes for a point (or corner, face) are stored in one block, which
39// is advantageous in terms of memory access. The length of the entire block is
40// given by the byte_stride, the position where the attribute starts is given by
41// the byte_offset, the actual number of bytes that the attribute occupies is
42// given by the data_type and the number of components.
43class GeometryAttribute {
44 public:
45 // Supported attribute types.
46 enum Type {
47 INVALID = -1,
48 // Named attributes start here. The difference between named and generic
49 // attributes is that for named attributes we know their purpose and we
50 // can apply some special methods when dealing with them (e.g. during
51 // encoding).
52 POSITION = 0,
53 NORMAL,
54 COLOR,
55 TEX_COORD,
56 // A special id used to mark attributes that are not assigned to any known
57 // predefined use case. Such attributes are often used for a shader specific
58 // data.
59 GENERIC,
60#ifdef DRACO_TRANSCODER_SUPPORTED
61 // TODO(ostava): Adding a new attribute would be bit-stream change for GLTF.
62 // Older decoders wouldn't know what to do with this attribute type. This
63 // should be open-sourced only when we are ready to increase our bit-stream
64 // version.
65 TANGENT,
66 MATERIAL,
67 JOINTS,
68 WEIGHTS,
69#endif
70 // Total number of different attribute types.
71 // Always keep behind all named attributes.
72 NAMED_ATTRIBUTES_COUNT,
73 };

87 // Function for getting a attribute value with a specific format.
88 // Unsafe. Caller must ensure the accessed memory is valid.
89 // T is the attribute data type.
90 // att_components_t is the number of attribute components.
91 template <typename T, int att_components_t>
92 std::array<T, att_components_t> GetValue(
93 AttributeValueIndex att_index) const {
94 // Byte address of the attribute index.
95 const int64_t byte_pos = byte_offset_ + byte_stride_ * att_index.value();
96 std::array<T, att_components_t> out;
97 buffer_->Read(byte_pos, &(out[0]), sizeof(out));
98 return out;
99 }
100
101 // Function for getting a attribute value with a specific format.
102 // T is the attribute data type.
103 // att_components_t is the number of attribute components.
104 template <typename T, int att_components_t>
105 bool GetValue(AttributeValueIndex att_index,
106 std::array<T, att_components_t> *out) const {
107 // Byte address of the attribute index.
108 const int64_t byte_pos = byte_offset_ + byte_stride_ * att_index.value();
109 // Check we are not reading past end of data.
110 if (byte_pos + sizeof(*out) > buffer_->data_size()) {
111 return false;
112 }
113 buffer_->Read(byte_pos, &((*out)[0]), sizeof(*out));
114 return true;
115 }
116
117 // Returns the byte position of the attribute entry in the data buffer.
118 inline int64_t GetBytePos(AttributeValueIndex att_index) const {
119 return byte_offset_ + byte_stride_ * att_index.value();
120 }
121
122 inline const uint8_t *GetAddress(AttributeValueIndex att_index) const {
123 const int64_t byte_pos = GetBytePos(att_index);
124 return buffer_->data() + byte_pos;
125 }
126 inline uint8_t *GetAddress(AttributeValueIndex att_index) {
127 const int64_t byte_pos = GetBytePos(att_index);
128 return buffer_->data() + byte_pos;
129 }
130 inline bool IsAddressValid(const uint8_t *address) const {
131 return ((buffer_->data() + buffer_->data_size()) > address);
132 }

173 // Function for conversion of a attribute to a specific output format.
174 // |out_val| needs to be able to store |out_num_components| values.
175 // OutT is the desired data type of the attribute.
176 // Returns false when the conversion failed.
177 template <typename OutT>
178 bool ConvertValue(AttributeValueIndex att_id, int8_t out_num_components,
179 OutT *out_val) const {
180 if (out_val == nullptr) {
181 return false;
182 }
183 switch (data_type_) {
184 case DT_INT8:
185 return ConvertTypedValue<int8_t, OutT>(att_id, out_num_components,
186 out_val);
187 case DT_UINT8:
188 return ConvertTypedValue<uint8_t, OutT>(att_id, out_num_components,
189 out_val);
190 case DT_INT16:
191 return ConvertTypedValue<int16_t, OutT>(att_id, out_num_components,
192 out_val);
193 case DT_UINT16:
194 return ConvertTypedValue<uint16_t, OutT>(att_id, out_num_components,
195 out_val);
196 case DT_INT32:
197 return ConvertTypedValue<int32_t, OutT>(att_id, out_num_components,
198 out_val);
199 case DT_UINT32:
200 return ConvertTypedValue<uint32_t, OutT>(att_id, out_num_components,
201 out_val);
202 case DT_INT64:
203 return ConvertTypedValue<int64_t, OutT>(att_id, out_num_components,
204 out_val);
205 case DT_UINT64:
206 return ConvertTypedValue<uint64_t, OutT>(att_id, out_num_components,
207 out_val);
208 case DT_FLOAT32:
209 return ConvertTypedValue<float, OutT>(att_id, out_num_components,
210 out_val);
211 case DT_FLOAT64:
212 return ConvertTypedValue<double, OutT>(att_id, out_num_components,
213 out_val);
214 case DT_BOOL:
215 return ConvertTypedValue<bool, OutT>(att_id, out_num_components,
216 out_val);
217 default:
218 // Wrong attribute type.
219 return false;
220 }
221 }

286 // Function for conversion of an attribute to a specific output format given a
287 // format of the stored attribute.
288 // T is the stored attribute data type.
289 // OutT is the desired data type of the attribute.
290 template <typename T, typename OutT>
291 bool ConvertTypedValue(AttributeValueIndex att_id, uint8_t out_num_components,
292 OutT *out_value) const {
293 const uint8_t *src_address = GetAddress(att_id);
294
295 // Convert all components available in both the original and output formats.
296 for (int i = 0; i < std::min(num_components_, out_num_components); ++i) {
297 if (!IsAddressValid(src_address)) {
298 return false;
299 }
300 const T in_value = *reinterpret_cast<const T *>(src_address);
301 if (!ConvertComponentValue<T, OutT>(in_value, normalized_,
302 out_value + i)) {
303 return false;
304 }
305 src_address += sizeof(T);
306 }
307 // Fill empty data for unused output components if needed.
308 for (int i = num_components_; i < out_num_components; ++i) {
309 out_value[i] = static_cast<OutT>(0);
310 }
311 return true;
312 }

314#ifdef DRACO_TRANSCODER_SUPPORTED
315 // Function that converts input |value| from type T to the internal attribute
316 // representation defined by OutT and |num_components_|.
317 template <typename T, typename OutT>
318 Status ConvertAndSetAttributeTypedValue(AttributeValueIndex avi,
319 int8_t input_num_components,
320 const T *value) {
321 uint8_t *address = GetAddress(avi);
322
323 // Convert all components available in both the original and output formats.
324 for (int i = 0; i < num_components_; ++i) {
325 if (!IsAddressValid(address)) {
326 return ErrorStatus("GeometryAttribute: Invalid address.");
327 }
328 OutT *const out_value = reinterpret_cast<OutT *>(address);
329 if (i < input_num_components) {
330 if (!ConvertComponentValue<T, OutT>(*(value + i), normalized_,
331 out_value)) {
332 return ErrorStatus(
333 "GeometryAttribute: Failed to convert component value.");
334 }
335 } else {
336 *out_value = static_cast<OutT>(0);
337 }
338 address += sizeof(OutT);
339 }
340 return OkStatus();
341 }
342#endif // DRACO_TRANSCODER_SUPPORTED

344 // Converts |in_value| of type T into |out_value| of type OutT. If
345 // |normalized| is true, any conversion between floating point and integer
346 // values will be treating integers as normalized types (the entire integer
347 // range will be used to represent 0-1 floating point range).
348 template <typename T, typename OutT>
349 static bool ConvertComponentValue(const T &in_value, bool normalized,
350 OutT *out_value) {
351 // Make sure the |in_value| can be represented as an integral type OutT.
352 if (std::is_integral<OutT>::value) {
353 // Make sure the |in_value| fits within the range of values that OutT
354 // is able to represent. Perform the check only for integral types.
355 if (!std::is_same<T, bool>::value && std::is_integral<T>::value) {
356 static constexpr OutT kOutMin =
357 std::is_signed<T>::value ? std::numeric_limits<OutT>::min() : 0;
358 if (in_value < kOutMin || in_value > std::numeric_limits<OutT>::max()) {
359 return false;
360 }
361 }
362
363 // Check conversion of floating point |in_value| to integral value OutT.
364 if (std::is_floating_point<T>::value) {
365 // Make sure the floating point |in_value| is not NaN and not Inf as
366 // integral type OutT is unable to represent these values.
367 if (sizeof(in_value) > sizeof(double)) {
368 if (std::isnan(static_cast<long double>(in_value)) ||
369 std::isinf(static_cast<long double>(in_value))) {
370 return false;
371 }
372 } else if (sizeof(in_value) > sizeof(float)) {
373 if (std::isnan(static_cast<double>(in_value)) ||
374 std::isinf(static_cast<double>(in_value))) {
375 return false;
376 }
377 } else {
378 if (std::isnan(static_cast<float>(in_value)) ||
379 std::isinf(static_cast<float>(in_value))) {
380 return false;
381 }
382 }
383
384 // Make sure the floating point |in_value| fits within the range of
385 // values that integral type OutT is able to represent.
386 if (in_value < std::numeric_limits<OutT>::min() ||
387 in_value >= std::numeric_limits<OutT>::max()) {
388 return false;
389 }
390 }
391 }
392
393 if (std::is_integral<T>::value && std::is_floating_point<OutT>::value &&
394 normalized) {
395 // When converting integer to floating point, normalize the value if
396 // necessary.
397 *out_value = static_cast<OutT>(in_value);
398 *out_value /= static_cast<OutT>(std::numeric_limits<T>::max());
399 } else if (std::is_floating_point<T>::value &&
400 std::is_integral<OutT>::value && normalized) {
401 // Converting from floating point to a normalized integer.
402 if (in_value > 1 || in_value < 0) {
403 // Normalized float values need to be between 0 and 1.
404 return false;
405 }
406 // TODO(ostava): Consider allowing float to normalized integer conversion
407 // for 64-bit integer types. Currently it doesn't work because we don't
408 // have a floating point type that could store all 64 bit integers.
409 if (sizeof(OutT) > 4) {
410 return false;
411 }
412 // Expand the float to the range of the output integer and round it to the
413 // nearest representable value. Use doubles for the math to ensure the
414 // integer values are represented properly during the conversion process.
415 *out_value = static_cast<OutT>(std::floor(
416 in_value * static_cast<double>(std::numeric_limits<OutT>::max()) +
417 0.5));
418 } else {
419 *out_value = static_cast<OutT>(in_value);
420 }
421
422 // TODO(ostava): Add handling of normalized attributes when converting
423 // between different integer representations. If the attribute is
424 // normalized, integer values should be converted as if they represent 0-1
425 // range. E.g. when we convert uint16 to uint8, the range <0, 2^16 - 1>
426 // should be converted to range <0, 2^8 - 1>.
427 return true;
428 }

501// Function object for using Attribute as a hash key.
502struct GeometryAttributeHasher {
503 size_t operator()(const GeometryAttribute &va) const {
504 size_t hash = HashCombine(va.buffer_descriptor_.buffer_id,
505 va.buffer_descriptor_.buffer_update_count);
506 hash = HashCombine(va.num_components_, hash);
507 hash = HashCombine(static_cast<int8_t>(va.data_type_), hash);
508 hash = HashCombine(static_cast<int8_t>(va.attribute_type_), hash);
509 hash = HashCombine(va.byte_stride_, hash);
510 return HashCombine(va.byte_offset_, hash);
511 }
512};
