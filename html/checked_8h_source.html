82 template <typename octet_iterator, typename output_iterator>
83 output_iterator replace_invalid(octet_iterator start, octet_iterator end, output_iterator out, uint32_t replacement)
84 {
85 while (start != end) {
86 octet_iterator sequence_start = start;
87 internal::utf_error err_code = utf8::internal::validate_next(start, end);
88 switch (err_code) {
89 case internal::UTF8_OK :
90 for (octet_iterator it = sequence_start; it != start; ++it)
91 *out++ = *it;
92 break;
93 case internal::NOT_ENOUGH_ROOM:
94 out = utf8::append (replacement, out);
95 start = end;
96 break;
97 case internal::INVALID_LEAD:
98 out = utf8::append (replacement, out);
99 ++start;
100 break;
101 case internal::INCOMPLETE_SEQUENCE:
102 case internal::OVERLONG_SEQUENCE:
103 case internal::INVALID_CODE_POINT:
104 out = utf8::append (replacement, out);
105 ++start;
106 // just one replacement mark for the sequence
107 while (start != end && utf8::internal::is_trail(*start))
108 ++start;
109 break;
110 }
111 }
112 return out;
113 }

122 template <typename octet_iterator>
123 uint32_t next(octet_iterator& it, octet_iterator end)
124 {
125 uint32_t cp = 0;
126 internal::utf_error err_code = utf8::internal::validate_next(it, end, cp);
127 switch (err_code) {
128 case internal::UTF8_OK :
129 break;
130 case internal::NOT_ENOUGH_ROOM :
131 throw not_enough_room();
132 case internal::INVALID_LEAD :
133 case internal::INCOMPLETE_SEQUENCE :
134 case internal::OVERLONG_SEQUENCE :
135 throw invalid_utf8(static_cast<uint8_t>(*it));
136 case internal::INVALID_CODE_POINT :
137 throw invalid_code_point(cp);
138 }
139 return cp;
140 }

148 template <typename octet_iterator>
149 uint32_t prior(octet_iterator& it, octet_iterator start)
150 {
151 // can't do much if it == start
152 if (it == start)
153 throw not_enough_room();
154
155 octet_iterator end = it;
156 // Go back until we hit either a lead octet or start
157 while (utf8::internal::is_trail(*(--it)))
158 if (it == start)
159 throw invalid_utf8(*it); // error - no lead byte in the sequence
160 return utf8::peek_next(it, end);
161 }

163 template <typename octet_iterator, typename distance_type>
164 void advance (octet_iterator& it, distance_type n, octet_iterator end)
165 {
166 const distance_type zero(0);
167 if (n < zero) {
168 // backward
169 for (distance_type i = n; i < zero; ++i)
170 utf8::prior(it, end);
171 } else {
172 // forward
173 for (distance_type i = zero; i < n; ++i)
174 utf8::next(it, end);
175 }
176 }

188 template <typename u16bit_iterator, typename octet_iterator>
189 octet_iterator utf16to8 (u16bit_iterator start, u16bit_iterator end, octet_iterator result)
190 {
191 while (start != end) {
192 uint32_t cp = utf8::internal::mask16(*start++);
193 // Take care of surrogate pairs first
194 if (utf8::internal::is_lead_surrogate(cp)) {
195 if (start != end) {
196 uint32_t trail_surrogate = utf8::internal::mask16(*start++);
197 if (utf8::internal::is_trail_surrogate(trail_surrogate))
198 cp = (cp << 10) + trail_surrogate + internal::SURROGATE_OFFSET;
199 else
200 throw invalid_utf16(static_cast<uint16_t>(trail_surrogate));
201 }
202 else
203 throw invalid_utf16(static_cast<uint16_t>(cp));
204
205 }
206 // Lone trail surrogate
207 else if (utf8::internal::is_trail_surrogate(cp))
208 throw invalid_utf16(static_cast<uint16_t>(cp));
209
210 result = utf8::append(cp, result);
211 }
212 return result;
213 }

215 template <typename u16bit_iterator, typename octet_iterator>
216 u16bit_iterator utf8to16 (octet_iterator start, octet_iterator end, u16bit_iterator result)
217 {
218 while (start < end) {
219 uint32_t cp = utf8::next(start, end);
220 if (cp > 0xffff) { //make a surrogate pair
221 *result++ = static_cast<uint16_t>((cp >> 10) + internal::LEAD_OFFSET);
222 *result++ = static_cast<uint16_t>((cp & 0x3ff) + internal::TRAIL_SURROGATE_MIN);
223 }
224 else
225 *result++ = static_cast<uint16_t>(cp);
226 }
227 return result;
228 }
