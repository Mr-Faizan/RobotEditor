242// versions. This code her is inlined, so it shouldn't cause any problems.

271 if ( path.empty() ) {

296// ----------------------------------------------------------------------------
297AI_FORCE_INLINE bool IOSystem::CreateDirectory( const std::string &path ) {
298 if ( path.empty() ) {
299 return false;
300 }
301
302#ifdef _WIN32
303 return 0 != ::_mkdir( path.c_str() );
304#else
305 return 0 != ::mkdir( path.c_str(), 0777 );
306#endif // _WIN32
307}

314
315#ifdef _WIN32

317#else
318 return 0 != ::chdir( path.c_str() );
319#endif // _WIN32
320}

325 if ( file.empty() ) {
326 return false;
327 }
328 const int retCode( ::remove( file.c_str() ) );
329 return ( 0 == retCode );
