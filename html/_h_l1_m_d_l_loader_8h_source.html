228template <typename MDLFileHeader>
229void HL1MDLLoader::load_file_into_buffer(const std::string &file_path, unsigned char *&buffer) {
230 if (!io_->Exists(file_path))
231 throw DeadlyImportError("Missing file ", DefaultIOSystem::fileName(file_path), ".");
232
233 std::unique_ptr<IOStream> file(io_->Open(file_path));
234
235 if (file == nullptr) {
236 throw DeadlyImportError("Failed to open MDL file ", DefaultIOSystem::fileName(file_path), ".");
237 }
238
239 const size_t file_size = file->FileSize();
240 if (file_size < sizeof(MDLFileHeader)) {
241 throw DeadlyImportError("MDL file is too small.");
242 }
243
244 buffer = new unsigned char[1 + file_size];
245 file->Read((void *)buffer, 1, file_size);
246 buffer[file_size] = '\0';
247}
