25// Class for holding parameter values for an attribute transform of a
26// PointAttribute. This can be for example quantization data for an attribute
27// that holds quantized values. This class provides only a basic storage for
28// attribute transform parameters and it should be accessed only through wrapper
29// classes for a specific transform (e.g. AttributeQuantizationTransform).
30class AttributeTransformData {
31 public:
32 AttributeTransformData() : transform_type_(ATTRIBUTE_INVALID_TRANSFORM) {}
33 AttributeTransformData(const AttributeTransformData &data) = default;
34
35 // Returns the type of the attribute transform that is described by the class.
36 AttributeTransformType transform_type() const { return transform_type_; }
37 void set_transform_type(AttributeTransformType type) {
38 transform_type_ = type;
39 }
40
41 // Returns a parameter value on a given |byte_offset|.
42 template <typename DataTypeT>
43 DataTypeT GetParameterValue(int byte_offset) const {
44 DataTypeT out_data;
45 buffer_.Read(byte_offset, &out_data, sizeof(DataTypeT));
46 return out_data;
47 }
48
49 // Sets a parameter value on a given |byte_offset|.
50 template <typename DataTypeT>
51 void SetParameterValue(int byte_offset, const DataTypeT &in_data) {
52 if (byte_offset + sizeof(DataTypeT) > buffer_.data_size()) {
53 buffer_.Resize(byte_offset + sizeof(DataTypeT));
54 }
55 buffer_.Write(byte_offset, &in_data, sizeof(DataTypeT));
56 }
57
58 // Sets a parameter value at the end of the |buffer_|.
59 template <typename DataTypeT>
60 void AppendParameterValue(const DataTypeT &in_data) {
61 SetParameterValue(static_cast<int>(buffer_.data_size()), in_data);
62 }
63
64 private:
65 AttributeTransformType transform_type_;
66 DataBuffer buffer_;
67};
