109template <class T>
110Ref<T> LazyDict<T>::Get(const char *id) {
111 id = T::TranslateId(mAsset, id);
112
113 typename Dict::iterator it = mObjsById.find(id);
114 if (it != mObjsById.end()) { // already created?
115 return Ref<T>(mObjs, it->second);
116 }
117
118 // read it from the JSON object
119 if (!mDict) {
120 throw DeadlyImportError("GLTF: Missing section \"", mDictId, "\"");
121 }
122
123 Value::MemberIterator obj = mDict->FindMember(id);
124 if (obj == mDict->MemberEnd()) {
125 throw DeadlyImportError("GLTF: Missing object with id \"", id, "\" in \"", mDictId, "\"");
126 }
127 if (!obj->value.IsObject()) {
128 throw DeadlyImportError("GLTF: Object with id \"", id, "\" is not a JSON object");
129 }
130
131 // create an instance of the given type
132 T *inst = new T();
133 inst->id = id;
134 ReadMember(obj->value, "name", inst->name);
135 inst->Read(obj->value, mAsset);
136 return Add(inst);
137}

139template <class T>
140Ref<T> LazyDict<T>::Add(T *obj) {
141 unsigned int idx = unsigned(mObjs.size());
142 mObjs.push_back(obj);
143 mObjsById[obj->id] = idx;
144 mAsset.mUsedIds[obj->id] = true;
145 return Ref<T>(mObjs, idx);
146}

148template <class T>
149Ref<T> LazyDict<T>::Create(const char *id) {
150 Asset::IdMap::iterator it = mAsset.mUsedIds.find(id);
151 if (it != mAsset.mUsedIds.end()) {
152 throw DeadlyImportError("GLTF: two objects with the same ID exist");
153 }
154 T *inst = new T();
155 inst->id = id;
156 return Add(inst);
157}

180inline void Buffer::Read(Value &obj, Asset &r) {
181 size_t statedLength = MemberOrDefault<size_t>(obj, "byteLength", 0);
182 byteLength = statedLength;
183
184 Value *it = FindString(obj, "uri");
185 if (!it) {
186 if (statedLength > 0) {
187 throw DeadlyImportError("GLTF: buffer with non-zero length missing the \"uri\" attribute");
188 }
189 return;
190 }
191
192 const char *uri = it->GetString();
193
194 glTFCommon::Util::DataURI dataURI;
195 if (ParseDataURI(uri, it->GetStringLength(), dataURI)) {
196 if (dataURI.base64) {
197 uint8_t *data = nullptr;
198 this->byteLength = Base64::Decode(dataURI.data, dataURI.dataLength, data);
199 this->mData.reset(data, std::default_delete<uint8_t[]>());
200
201 if (statedLength > 0 && this->byteLength != statedLength) {
202 throw DeadlyImportError("GLTF: buffer \"", id, "\", expected ", ai_to_string(statedLength),
203 " bytes, but found ", ai_to_string(dataURI.dataLength));
204 }
205 } else { // assume raw data
206 if (statedLength != dataURI.dataLength) {
207 throw DeadlyImportError("GLTF: buffer \"", id, "\", expected ", ai_to_string(statedLength),
208 " bytes, but found ", ai_to_string(dataURI.dataLength));
209 }
210
211 this->mData.reset(new uint8_t[dataURI.dataLength], std::default_delete<uint8_t[]>());
212 memcpy(this->mData.get(), dataURI.data, dataURI.dataLength);
213 }
214 } else { // Local file
215 if (byteLength > 0) {
216 std::string dir = !r.mCurrentAssetDir.empty() ? (
217 r.mCurrentAssetDir.back() == '/' ?
218 r.mCurrentAssetDir :
219 r.mCurrentAssetDir + '/') :
220 "";
221
222 IOStream *file = r.OpenFile(dir + uri, "rb");
223 if (file) {
224 bool ok = LoadFromStream(*file, byteLength);
225 delete file;
226
227 if (!ok)
228 throw DeadlyImportError("GLTF: error while reading referenced file \"", uri, "\"");
229 } else {
230 throw DeadlyImportError("GLTF: could not open referenced file \"", uri, "\"");
231 }
232 }
233 }

236inline bool Buffer::LoadFromStream(IOStream &stream, size_t length, size_t baseOffset) {
237 byteLength = length ? length : stream.FileSize();
238
239 if (baseOffset) {
240 stream.Seek(baseOffset, aiOrigin_SET);
241 }
242
243 mData.reset(new uint8_t[byteLength], std::default_delete<uint8_t[]>());
244
245 if (stream.Read(mData.get(), byteLength, 1) != 1) {
246 return false;
247 }
248 return true;
249}

251inline void Buffer::EncodedRegion_Mark(const size_t pOffset, const size_t pEncodedData_Length, uint8_t *pDecodedData, const size_t pDecodedData_Length, const std::string &pID) {
252 // Check pointer to data
253 if (pDecodedData == nullptr) throw DeadlyImportError("GLTF: for marking encoded region pointer to decoded data must be provided.");
254
255 // Check offset
256 if (pOffset > byteLength) {
257 const uint8_t val_size = 32;
258
259 char val[val_size];
260
261 ai_snprintf(val, val_size, AI_SIZEFMT, pOffset);
262 throw DeadlyImportError("GLTF: incorrect offset value (", val, ") for marking encoded region.");
263 }
264
265 // Check length
266 if ((pOffset + pEncodedData_Length) > byteLength) {
267 const uint8_t val_size = 64;
268
269 char val[val_size];
270
271 ai_snprintf(val, val_size, AI_SIZEFMT "/" AI_SIZEFMT, pOffset, pEncodedData_Length);
272 throw DeadlyImportError("GLTF: encoded region with offset/length (", val, ") is out of range.");
273 }
274
275 // Add new region
276 EncodedRegion_List.push_back(new SEncodedRegion(pOffset, pEncodedData_Length, pDecodedData, pDecodedData_Length, pID));
277 // And set new value for "byteLength"
278 byteLength += (pDecodedData_Length - pEncodedData_Length);
279}

281inline void Buffer::EncodedRegion_SetCurrent(const std::string &pID) {
282 if ((EncodedRegion_Current != nullptr) && (EncodedRegion_Current->ID == pID)) return;
283
284 for (SEncodedRegion *reg : EncodedRegion_List) {
285 if (reg->ID == pID) {
286 EncodedRegion_Current = reg;

288 return;
289 }
290 }
291
292 throw DeadlyImportError("GLTF: EncodedRegion with ID: \"", pID, "\" not found.");

295inline bool Buffer::ReplaceData(const size_t pBufferData_Offset, const size_t pBufferData_Count, const uint8_t *pReplace_Data, const size_t pReplace_Count) {
296 const size_t new_data_size = byteLength + pReplace_Count - pBufferData_Count;
297
298 uint8_t *new_data;
299
300 if ((pBufferData_Count == 0) || (pReplace_Count == 0) || (pReplace_Data == nullptr)) return false;
301
302 new_data = new uint8_t[new_data_size];
303 // Copy data which place before replacing part.
304 memcpy(new_data, mData.get(), pBufferData_Offset);
305 // Copy new data.
306 memcpy(&new_data[pBufferData_Offset], pReplace_Data, pReplace_Count);
307 // Copy data which place after replacing part.
308 memcpy(&new_data[pBufferData_Offset + pReplace_Count], &mData.get()[pBufferData_Offset + pBufferData_Count], pBufferData_Offset);
309 // Apply new data
310 mData.reset(new_data, std::default_delete<uint8_t[]>());
311 byteLength = new_data_size;
312
313 return true;
314}

316inline size_t Buffer::AppendData(uint8_t *data, size_t length) {
317 size_t offset = this->byteLength;
318 Grow(length);
319 memcpy(mData.get() + offset, data, length);
320 return offset;
321}

323inline void Buffer::Grow(size_t amount) {
324 if (amount <= 0) return;
325 if (capacity >= byteLength + amount) {
326 byteLength += amount;
327 return;
328 }
329
330 // Shift operation is standard way to divide integer by 2, it doesn't cast it to float back and forth, also works for odd numbers,
331 // originally it would look like: static_cast<size_t>(capacity * 1.5f)
332 capacity = std::max(capacity + (capacity >> 1), byteLength + amount);
333
334 uint8_t *b = new uint8_t[capacity];
335 if (mData) memcpy(b, mData.get(), byteLength);
336 mData.reset(b, std::default_delete<uint8_t[]>());
337 byteLength += amount;
338}

385inline uint8_t *Accessor::GetPointer() {
386 if (!bufferView || !bufferView->buffer) return nullptr;
387 uint8_t *basePtr = bufferView->buffer->GetPointer();
388 if (!basePtr) return nullptr;
389
390 size_t offset = byteOffset + bufferView->byteOffset;
391
392 // Check if region is encoded.
393 if (bufferView->buffer->EncodedRegion_Current != nullptr) {
394 const size_t begin = bufferView->buffer->EncodedRegion_Current->Offset;
395 const size_t end = begin + bufferView->buffer->EncodedRegion_Current->DecodedData_Length;
396
397 if ((offset >= begin) && (offset < end))
398 return &bufferView->buffer->EncodedRegion_Current->DecodedData[offset - begin];
399 }
400
401 return basePtr + offset;
402}

405inline void CopyData(size_t count,
406 const uint8_t *src, size_t src_stride,
407 uint8_t *dst, size_t dst_stride) {
408 if (src_stride == dst_stride) {
409 memcpy(dst, src, count * src_stride);
410 } else {
411 size_t sz = std::min(src_stride, dst_stride);
412 for (size_t i = 0; i < count; ++i) {
413 memcpy(dst, src, sz);
414 if (sz < dst_stride) {
415 memset(dst + sz, 0, dst_stride - sz);
416 }
417 src += src_stride;
418 dst += dst_stride;
419 }
420 }
421}

424template <class T>
425bool Accessor::ExtractData(T *&outData) {
426 uint8_t *data = GetPointer();
427 if (!data) return false;
428
429 const size_t elemSize = GetElementSize();
430 const size_t totalSize = elemSize * count;
431
432 const size_t stride = byteStride ? byteStride : elemSize;
433
434 const size_t targetElemSize = sizeof(T);
435 ai_assert(elemSize <= targetElemSize);
436
437 ai_assert(count * stride <= bufferView->byteLength);
438
439 outData = new T[count];
440 if (stride == elemSize && targetElemSize == elemSize) {
441 memcpy(outData, data, totalSize);
442 } else {
443 for (size_t i = 0; i < count; ++i) {
444 memcpy(outData + i, data + i * stride, elemSize);
445 }
446 }
447
448 return true;
449}

451inline void Accessor::WriteData(size_t cnt, const void *src_buffer, size_t src_stride) {
452 uint8_t *buffer_ptr = bufferView->buffer->GetPointer();
453 size_t offset = byteOffset + bufferView->byteOffset;
454
455 size_t dst_stride = GetNumComponents() * GetBytesPerComponent();
456
457 const uint8_t *src = reinterpret_cast<const uint8_t *>(src_buffer);
458 uint8_t *dst = reinterpret_cast<uint8_t *>(buffer_ptr + offset);
459
460 ai_assert(dst + count * dst_stride <= buffer_ptr + bufferView->buffer->byteLength);
461 CopyData(cnt, src, src_stride, dst, dst_stride);
462}

483inline void Image::Read(Value &obj, Asset &r) {
484 // Check for extensions first (to detect binary embedded data)
485 if (Value *extensions = FindObject(obj, "extensions")) {
486 if (r.extensionsUsed.KHR_binary_glTF) {
487 if (Value *ext = FindObject(*extensions, "KHR_binary_glTF")) {
488
489 width = MemberOrDefault(*ext, "width", 0);
490 height = MemberOrDefault(*ext, "height", 0);
491
492 ReadMember(*ext, "mimeType", mimeType);
493
494 const char *bufferViewId;
495 if (ReadMember(*ext, "bufferView", bufferViewId)) {
496 Ref<BufferView> bv = r.bufferViews.Get(bufferViewId);
497 if (bv) {
498 mDataLength = bv->byteLength;
499 mData.reset(new uint8_t[mDataLength]);
500 memcpy(mData.get(), bv->buffer->GetPointer() + bv->byteOffset, mDataLength);
501 }
502 }
503 }
504 }
505 }
506
507 if (!mDataLength) {
508 Value *curUri = FindString(obj, "uri");
509 if (nullptr != curUri) {
510 const char *uristr = curUri->GetString();
511
512 glTFCommon::Util::DataURI dataURI;
513 if (ParseDataURI(uristr, curUri->GetStringLength(), dataURI)) {
514 mimeType = dataURI.mediaType;
515 if (dataURI.base64) {
516 uint8_t *ptr = nullptr;
517 mDataLength = Base64::Decode(dataURI.data, dataURI.dataLength, ptr);
518 mData.reset(ptr);
519 }
520 } else {
521 this->uri = uristr;
522 }
523 }
524 }

682inline void Mesh::Read(Value &pJSON_Object, Asset &pAsset_Root) {
683 /****************** Mesh primitives ******************/
684 Value *curPrimitives = FindArray(pJSON_Object, "primitives");
685 if (nullptr != curPrimitives) {
686 this->primitives.resize(curPrimitives->Size());
687 for (unsigned int i = 0; i < curPrimitives->Size(); ++i) {
688 Value &primitive = (*curPrimitives)[i];
689
690 Primitive &prim = this->primitives[i];
691 prim.mode = MemberOrDefault(primitive, "mode", PrimitiveMode_TRIANGLES);
692
693 if (Value *attrs = FindObject(primitive, "attributes")) {
694 for (Value::MemberIterator it = attrs->MemberBegin(); it != attrs->MemberEnd(); ++it) {
695 if (!it->value.IsString()) continue;
696 const char *attr = it->name.GetString();
697 // Valid attribute semantics include POSITION, NORMAL, TEXCOORD, COLOR, JOINT, JOINTMATRIX,
698 // and WEIGHT.Attribute semantics can be of the form[semantic]_[set_index], e.g., TEXCOORD_0, TEXCOORD_1, etc.
699
700 int undPos = 0;
701 Mesh::AccessorList *vec = nullptr;
702 if (GetAttribVector(prim, attr, vec, undPos)) {
703 size_t idx = (attr[undPos] == '_') ? atoi(attr + undPos + 1) : 0;
704 if ((*vec).size() <= idx) (*vec).resize(idx + 1);
705 (*vec)[idx] = pAsset_Root.accessors.Get(it->value.GetString());
706 }
707 }
708 }
709
710 if (Value *indices = FindString(primitive, "indices")) {
711 prim.indices = pAsset_Root.accessors.Get(indices->GetString());
712 }
713
714 if (Value *material = FindString(primitive, "material")) {
715 prim.material = pAsset_Root.materials.Get(material->GetString());
716 }
717 }
718 }
719

721 Value *json_extensions = FindObject(pJSON_Object, "extensions");
722
723 if (json_extensions == nullptr) goto mr_skip_extensions;
724
725#ifdef ASSIMP_IMPORTER_GLTF_USE_OPEN3DGC
726 for (Value::MemberIterator it_memb = json_extensions->MemberBegin(); it_memb != json_extensions->MemberEnd(); it_memb++) {

738#define MESH_READ_COMPRESSEDDATA_MEMBER(pFieldName, pOut) \
739 if (!ReadMember(*comp_data, pFieldName, pOut)) { \
740 throw DeadlyImportError("GLTF: \"compressedData\" must has \"", pFieldName, "\"."); \

744 const char *type_str;
745 ComponentType component_type;

747
748 MESH_READ_COMPRESSEDDATA_MEMBER("buffer", ext_o3dgc->Buffer);
749 MESH_READ_COMPRESSEDDATA_MEMBER("byteOffset", ext_o3dgc->Offset);
750 MESH_READ_COMPRESSEDDATA_MEMBER("componentType", component_type);
751 MESH_READ_COMPRESSEDDATA_MEMBER("type", type_str);
752 MESH_READ_COMPRESSEDDATA_MEMBER("count", ext_o3dgc->Count);
753 MESH_READ_COMPRESSEDDATA_MEMBER("mode", mode_str);
754 MESH_READ_COMPRESSEDDATA_MEMBER("indicesCount", ext_o3dgc->IndicesCount);
755 MESH_READ_COMPRESSEDDATA_MEMBER("verticesCount", ext_o3dgc->VerticesCount);
756
757#undef MESH_READ_COMPRESSEDDATA_MEMBER
758
759 // Check some values
760 if (strcmp(type_str, "SCALAR")) throw DeadlyImportError("GLTF: only \"SCALAR\" type is supported for compressed data.");
761 if (component_type != ComponentType_UNSIGNED_BYTE) throw DeadlyImportError("GLTF: only \"UNSIGNED_BYTE\" component type is supported for compressed data.");
762
763 // Set read/write data mode.
764 if (strcmp(mode_str, "binary") == 0)
765 ext_o3dgc->Binary = true;
766 else if (strcmp(mode_str, "ascii") == 0)
767 ext_o3dgc->Binary = false;
768 else
769 throw DeadlyImportError("GLTF: for compressed data supported modes is: \"ascii\", \"binary\". Not the: \"", mode_str, "\".");
770
771 /************************ Decoding ************************/
772 Decode_O3DGC(*ext_o3dgc, pAsset_Root);
773 Extension.push_back(ext_o3dgc); // store info in mesh extensions list.
774 } // if(it_memb->name.GetString() == "Open3DGC-compression")
775 else {
776 throw DeadlyImportError("GLTF: Unknown mesh extension: \"", it_memb->name.GetString(), "\".");
777 }
778 } // for(Value::MemberIterator it_memb = json_extensions->MemberBegin(); it_memb != json_extensions->MemberEnd(); json_extensions++)
779#endif
780
781mr_skip_extensions:
782
783 return; // After label some operators must be present.
784}

787inline void Mesh::Decode_O3DGC(const SCompression_Open3DGC &pCompression_Open3DGC, Asset &pAsset_Root) {
788 typedef unsigned short IndicesType;
789
790 o3dgc::SC3DMCDecoder<IndicesType> decoder;
791 o3dgc::IndexedFaceSet<IndicesType> ifs;
792 o3dgc::BinaryStream bstream;
793 uint8_t *decoded_data;
794 size_t decoded_data_size = 0;
795 Ref<Buffer> buf = pAsset_Root.buffers.Get(pCompression_Open3DGC.Buffer);
796
797 // Read data from buffer and place it in BinaryStream for decoder.
798 // Just "Count" because always is used type equivalent to uint8_t.
799 bstream.LoadFromBuffer(&buf->GetPointer()[pCompression_Open3DGC.Offset], static_cast<unsigned long>(pCompression_Open3DGC.Count));
800
801 // After decoding header we can get size of primitives.
802 if (decoder.DecodeHeader(ifs, bstream) != o3dgc::O3DGC_OK) throw DeadlyImportError("GLTF: can not decode Open3DGC header.");
803
804 /****************** Get sizes of arrays and check sizes ******************/
805 // Note. See "Limitations for meshes when using Open3DGC-compression".
806
807 // Indices
808 size_t size_coordindex = ifs.GetNCoordIndex() * 3; // See float attributes note.
809
810 if (primitives[0].indices->count != size_coordindex)
811 throw DeadlyImportError("GLTF: Open3DGC. Compressed indices count (", ai_to_string(size_coordindex),
812 ") not equal to uncompressed (", ai_to_string(primitives[0].indices->count), ").");
813
814 size_coordindex *= sizeof(IndicesType);
815 // Coordinates
816 size_t size_coord = ifs.GetNCoord(); // See float attributes note.
817
818 if (primitives[0].attributes.position[0]->count != size_coord)
819 throw DeadlyImportError("GLTF: Open3DGC. Compressed positions count (", ai_to_string(size_coord),
820 ") not equal to uncompressed (", ai_to_string(primitives[0].attributes.position[0]->count), ").");
821
822 size_coord *= 3 * sizeof(float);
823 // Normals
824 size_t size_normal = ifs.GetNNormal(); // See float attributes note.
825
826 if (primitives[0].attributes.normal[0]->count != size_normal)
827 throw DeadlyImportError("GLTF: Open3DGC. Compressed normals count (", ai_to_string(size_normal),
828 ") not equal to uncompressed (", ai_to_string(primitives[0].attributes.normal[0]->count), ").");
829
830 size_normal *= 3 * sizeof(float);
831 // Additional attributes.
832 std::vector<size_t> size_floatattr;
833 std::vector<size_t> size_intattr;
834
835 size_floatattr.resize(ifs.GetNumFloatAttributes());
836 size_intattr.resize(ifs.GetNumIntAttributes());
837
838 decoded_data_size = size_coordindex + size_coord + size_normal;
839 for (size_t idx = 0, idx_end = size_floatattr.size(), idx_texcoord = 0; idx < idx_end; idx++) {
840 // size = number_of_elements * components_per_element * size_of_component.
841 // Note. But as you can see above, at first we are use this variable in meaning "count". After checking count of objects...
842 size_t tval = ifs.GetNFloatAttribute(static_cast<unsigned long>(idx));
843
844 switch (ifs.GetFloatAttributeType(static_cast<unsigned long>(idx))) {
845 case o3dgc::O3DGC_IFS_FLOAT_ATTRIBUTE_TYPE_TEXCOORD:
846 // Check situation when encoded data contain texture coordinates but primitive not.
847 if (idx_texcoord < primitives[0].attributes.texcoord.size()) {
848 if (primitives[0].attributes.texcoord[idx]->count != tval)
849 throw DeadlyImportError("GLTF: Open3DGC. Compressed texture coordinates count (", ai_to_string(tval),
850 ") not equal to uncompressed (", ai_to_string(primitives[0].attributes.texcoord[idx]->count), ").");
851
852 idx_texcoord++;
853 } else {
854 ifs.SetNFloatAttribute(static_cast<unsigned long>(idx), 0ul); // Disable decoding this attribute.
855 }
856
857 break;
858 default:
859 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of float attribute: ", ai_to_string(ifs.GetFloatAttributeType(static_cast<unsigned long>(idx))));
860 }
861
862 tval *= ifs.GetFloatAttributeDim(static_cast<unsigned long>(idx)) * sizeof(o3dgc::Real); // After checking count of objects we can get size of array.
863 size_floatattr[idx] = tval;
864 decoded_data_size += tval;
865 }
866
867 for (size_t idx = 0, idx_end = size_intattr.size(); idx < idx_end; idx++) {
868 // size = number_of_elements * components_per_element * size_of_component. See float attributes note.
869 size_t tval = ifs.GetNIntAttribute(static_cast<unsigned long>(idx));
870 switch (ifs.GetIntAttributeType(static_cast<unsigned long>(idx))) {
871 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_UNKOWN:
872 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX:
873 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_JOINT_ID:
874 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX_BUFFER_ID:
875 break;
876
877 default:
878 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of int attribute: ", ai_to_string(ifs.GetIntAttributeType(static_cast<unsigned long>(idx))));
879 }
880
881 tval *= ifs.GetIntAttributeDim(static_cast<unsigned long>(idx)) * sizeof(long); // See float attributes note.
882 size_intattr[idx] = tval;
883 decoded_data_size += tval;
884 }
885
886 // Create array for decoded data.
887 decoded_data = new uint8_t[decoded_data_size];
888
889 /****************** Set right array regions for decoder ******************/
890
891 auto get_buf_offset = [](Ref<Accessor> &pAccessor) -> size_t { return pAccessor->byteOffset + pAccessor->bufferView->byteOffset; };
892
893 // Indices
894 ifs.SetCoordIndex((IndicesType *)(decoded_data + get_buf_offset(primitives[0].indices)));
895 // Coordinates
896 ifs.SetCoord((o3dgc::Real *)(decoded_data + get_buf_offset(primitives[0].attributes.position[0])));
897 // Normals
898 if (size_normal) {
899 ifs.SetNormal((o3dgc::Real *)(decoded_data + get_buf_offset(primitives[0].attributes.normal[0])));
900 }
901
902 for (size_t idx = 0, idx_end = size_floatattr.size(), idx_texcoord = 0; idx < idx_end; idx++) {
903 switch (ifs.GetFloatAttributeType(static_cast<unsigned long>(idx))) {
904 case o3dgc::O3DGC_IFS_FLOAT_ATTRIBUTE_TYPE_TEXCOORD:
905 if (idx_texcoord < primitives[0].attributes.texcoord.size()) {
906 // See above about absent attributes.
907 ifs.SetFloatAttribute(static_cast<unsigned long>(idx), (o3dgc::Real *)(decoded_data + get_buf_offset(primitives[0].attributes.texcoord[idx])));
908 idx_texcoord++;
909 }
910
911 break;
912 default:
913 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of float attribute: ", ai_to_string(ifs.GetFloatAttributeType(static_cast<unsigned long>(idx))));
914 }
915 }
916
917 for (size_t idx = 0, idx_end = size_intattr.size(); idx < idx_end; idx++) {
918 switch (ifs.GetIntAttributeType(static_cast<unsigned int>(idx))) {
919 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_UNKOWN:
920 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX:
921 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_JOINT_ID:
922 case o3dgc::O3DGC_IFS_INT_ATTRIBUTE_TYPE_INDEX_BUFFER_ID:
923 break;
924
925 // ifs.SetIntAttribute(idx, (long* const)(decoded_data + get_buf_offset(primitives[0].attributes.joint)));
926 default:
927 throw DeadlyImportError("GLTF: Open3DGC. Unsupported type of int attribute: ", ai_to_string(ifs.GetIntAttributeType(static_cast<unsigned long>(idx))));
928 }
929 }
930
931 //
932 // Decode data
933 //
934 if (decoder.DecodePayload(ifs, bstream) != o3dgc::O3DGC_OK) {
935 throw DeadlyImportError("GLTF: can not decode Open3DGC data.");
936 }
937
938 // Set encoded region for "buffer".
939 buf->EncodedRegion_Mark(pCompression_Open3DGC.Offset, pCompression_Open3DGC.Count, decoded_data, decoded_data_size, id);
940 // No. Do not delete "output_data". After calling "EncodedRegion_Mark" bufferView is owner of "output_data".
941 // "delete [] output_data;"
942}

1123inline void Asset::ReadBinaryHeader(IOStream &stream) {
1124 GLB_Header header;
1125 if (stream.Read(&header, sizeof(header), 1) != 1) {
1126 throw DeadlyImportError("GLTF: Unable to read the file header");
1127 }
1128
1129 if (strncmp((char *)header.magic, AI_GLB_MAGIC_NUMBER, sizeof(header.magic)) != 0) {
1130 throw DeadlyImportError("GLTF: Invalid binary glTF file");
1131 }
1132
1133 AI_SWAP4(header.version);
1134 asset.version = ai_to_string(header.version);
1135 if (header.version != 1) {
1136 throw DeadlyImportError("GLTF: Unsupported binary glTF version");
1137 }
1138
1139 AI_SWAP4(header.sceneFormat);
1140 if (header.sceneFormat != SceneFormat_JSON) {
1141 throw DeadlyImportError("GLTF: Unsupported binary glTF scene format");
1142 }
1143
1144 AI_SWAP4(header.length);
1145 AI_SWAP4(header.sceneLength);
1146
1147 static_assert(std::numeric_limits<uint32_t>::max() <= std::numeric_limits<size_t>::max(), "size_t must be at least 32bits");
1148 mSceneLength = static_cast<size_t>(header.sceneLength); // Can't be larger than 4GB (max. uint32_t)
1149
1150 mBodyOffset = sizeof(header) + mSceneLength;
1151 mBodyOffset = (mBodyOffset + 3) & ~3; // Round up to next multiple of 4
1152
1153 mBodyLength = header.length - mBodyOffset;
1154}

1156inline void Asset::Load(const std::string &pFile, bool isBinary) {
1157 mCurrentAssetDir.clear();
1158
1159 /*int pos = std::max(int(pFile.rfind('/')), int(pFile.rfind('\\')));
1160 if (pos != int(std::string::npos)) mCurrentAssetDir = pFile.substr(0, pos + 1);*/
1161 if (0 != strncmp(pFile.c_str(), AI_MEMORYIO_MAGIC_FILENAME, AI_MEMORYIO_MAGIC_FILENAME_LENGTH)) {
1162 mCurrentAssetDir = getCurrentAssetDir(pFile);
1163 }
1164
1165 shared_ptr<IOStream> stream(OpenFile(pFile.c_str(), "rb", true));
1166 if (!stream) {
1167 throw DeadlyImportError("GLTF: Could not open file for reading");
1168 }
1169
1170 // is binary? then read the header
1171 if (isBinary) {
1172 SetAsBinary(); // also creates the body buffer
1173 ReadBinaryHeader(*stream);
1174 } else {
1175 mSceneLength = stream->FileSize();
1176 mBodyLength = 0;
1177 }
1178
1179 // Smallest legal JSON file is "{}" Smallest loadable glTF file is larger than that but catch it later
1180 if (mSceneLength < 2) {
1181 throw DeadlyImportError("GLTF: No JSON file contents");
1182 }
1183
1184 // Binary format only supports up to 4GB of JSON so limit it there to avoid extreme memory allocation
1185 if (mSceneLength >= std::numeric_limits<uint32_t>::max()) {
1186 throw DeadlyImportError("GLTF: JSON size greater than 4GB");
1187 }
1188
1189 // read the scene data, ensure null termination
1190 std::vector<char> sceneData(mSceneLength + 1);
1191 sceneData[mSceneLength] = '\0';
1192
1193 if (stream->Read(&sceneData[0], 1, mSceneLength) != mSceneLength) {
1194 throw DeadlyImportError("GLTF: Could not read the file contents");
1195 }
1196
1197 // parse the JSON document
1198
1199 Document doc;
1200 doc.ParseInsitu(&sceneData[0]);
1201
1202 if (doc.HasParseError()) {
1203 char buffer[32];
1204 ai_snprintf(buffer, 32, "%d", static_cast<int>(doc.GetErrorOffset()));
1205 throw DeadlyImportError("GLTF: JSON parse error, offset ", buffer, ": ", GetParseError_En(doc.GetParseError()));
1206 }
1207
1208 if (!doc.IsObject()) {
1209 throw DeadlyImportError("GLTF: JSON document root must be a JSON object");
1210 }
1211
1212 // Fill the buffer instance for the current file embedded contents
1213 if (mBodyLength > 0) {
1214 if (!mBodyBuffer->LoadFromStream(*stream, mBodyLength, mBodyOffset)) {
1215 throw DeadlyImportError("GLTF: Unable to read gltf file");
1216 }
1217 }
1218
1219 // Load the metadata
1220 asset.Read(doc);
1221 if (!asset) {
1222 return;
1223 }
1224
1225 ReadExtensionsUsed(doc);
1226
1227 // Prepare the dictionaries
1228 for (size_t i = 0; i < mDicts.size(); ++i) {
1229 mDicts[i]->AttachToDocument(doc);
1230 }
1231
1232 // Read the "scene" property, which specifies which scene to load
1233 // and recursively load everything referenced by it
1234 Value *curScene = FindString(doc, "scene");
1235 if (nullptr != curScene) {
1236 this->scene = scenes.Get(curScene->GetString());
1237 }
1238
1239 // Clean up
1240 for (size_t i = 0; i < mDicts.size(); ++i) {
1241 mDicts[i]->DetachFromDocument();
1242 }
1243}

1253inline void Asset::ReadExtensionsUsed(Document &doc) {
1254 Value *extsUsed = FindArray(doc, "extensionsUsed");
1255 if (!extsUsed) return;
1256
1257 std::gltf_unordered_map<std::string, bool> exts;
1258
