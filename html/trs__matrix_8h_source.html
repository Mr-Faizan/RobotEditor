26// This class is used to store one or more of a translation, rotation, scale
27// vectors or a transformation matrix.
28class TrsMatrix {
29 public:
30 TrsMatrix()
31 : matrix_(Eigen::Matrix4d::Identity()),
32 translation_(0.0, 0.0, 0.0),
33 rotation_(1.0, 0.0, 0.0, 0.0), // (w, x, y, z)
34 scale_(1.0, 1.0, 1.0),
35 matrix_set_(false),
36 translation_set_(false),
37 rotation_set_(false),
38 scale_set_(false) {}
39
40 void Copy(const TrsMatrix &tm);
41
42 void SetMatrix(const Eigen::Matrix4d &matrix) {
43 matrix_ = matrix;
44 matrix_set_ = true;
45 }
46 bool MatrixSet() const { return matrix_set_; }
47 StatusOr<const Eigen::Matrix4d> Matrix() const {
48 if (!matrix_set_) {
49 return Status(Status::DRACO_ERROR, "Matrix is not set.");
50 }
51 return matrix_;
52 }
53
54 void SetTranslation(const Eigen::Vector3d &translation) {
55 translation_ = translation;
56 translation_set_ = true;
57 }
58 bool TranslationSet() const { return translation_set_; }
59 StatusOr<const Eigen::Vector3d> Translation() const {
60 if (!translation_set_) {
61 return Status(Status::DRACO_ERROR, "Translation is not set.");
62 }
63 return translation_;
64 }
65
66 void SetRotation(const Eigen::Quaterniond &rotation) {
67 rotation_ = rotation;
68 rotation_set_ = true;
69 }
70 bool RotationSet() const { return rotation_set_; }
71 StatusOr<const Eigen::Quaterniond> Rotation() const {
72 if (!rotation_set_) {
73 return Status(Status::DRACO_ERROR, "Rotation is not set.");
74 }
75 return rotation_;
76 }
77
78 void SetScale(const Eigen::Vector3d &scale) {
79 scale_ = scale;
80 scale_set_ = true;
81 }
82 bool ScaleSet() const { return scale_set_; }
83 StatusOr<const Eigen::Vector3d> Scale() const {
84 if (!scale_set_) {
85 return Status(Status::DRACO_ERROR, "Scale is not set.");
86 }
87 return scale_;
88 }
89
90 // Returns true if the matrix is not set or if matrix is set and is equal to
91 // identity.
92 bool IsMatrixIdentity() const;
93
94 // Returns true if matrix is set and only the translation elements may differ
95 // from identity. Returns false if matrix is not set.
96 bool IsMatrixTranslationOnly() const;
97
98 // Returns transformation matrix if it has been set. Otherwise, computes
99 // transformation matrix from TRS vectors and returns it.
100 Eigen::Matrix4d ComputeTransformationMatrix() const;
101
102 // Returns a boolean indicating whether any of the transforms have been set.
103 // Can be used to check whether this object represents a default transform.
104 bool TransformSet() const {
105 return matrix_set_ || translation_set_ || rotation_set_ || scale_set_;
106 }
107
108 bool operator==(const TrsMatrix &trs_matrix) const;
109
110 private:

113 Eigen::Quaterniond rotation_;
114 Eigen::Vector3d scale_;
115 bool matrix_set_;
116 bool translation_set_;
117 bool rotation_set_;
118 bool scale_set_;
119};
