67struct Material : public D3DS::Material {
69 Material() = delete;
70
72 explicit Material(const std::string &name) :
73 D3DS::Material(name),
74 pcInstance(nullptr),
75 bNeed(false) {
76 // empty
77 }
78
79 Material(const Material &other) = default;
80
81 Material &operator=(const Material &other) {
82 if (this == &other) {
83 return *this;
84 }
85
86 avSubMaterials = other.avSubMaterials;
87 pcInstance = other.pcInstance;
88 bNeed = other.bNeed;
89
90 return *this;
91 }
92
94 Material(Material &&other) AI_NO_EXCEPT
95 : D3DS::Material(std::move(other)),
96 avSubMaterials(std::move(other.avSubMaterials)),
97 pcInstance(other.pcInstance),
98 bNeed(other.bNeed) {
99 other.pcInstance = nullptr;
100 }
101
102 Material &operator=(Material &&other) AI_NO_EXCEPT {
103 if (this == &other) {
104 return *this;
105 }
106
107 //D3DS::Material::operator=(std::move(other));
108
109 avSubMaterials = std::move(other.avSubMaterials);
110 pcInstance = other.pcInstance;
111 bNeed = other.bNeed;
112
113 other.pcInstance = nullptr;
114
115 return *this;
116 }
117
118 ~Material() = default;
119
121 std::vector<Material> avSubMaterials;
122
124 aiMaterial *pcInstance;
125
127 bool bNeed;
128};
