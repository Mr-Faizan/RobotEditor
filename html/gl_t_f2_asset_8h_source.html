322 }

333struct CustomExtension {
334
335 //
336 // A struct containing custom extension data added to a glTF2 file
337 // Has to contain Object, Array, String, Double, Uint64, and Int64 at a minimum
338 // String, Double, Uint64, and Int64 are stored in the Nullables
339 // Object and Array are stored in the std::vector
340 //
341 std::string name;
342
343 Nullable<std::string> mStringValue;
344 Nullable<double> mDoubleValue;
345 Nullable<uint64_t> mUint64Value;
346 Nullable<int64_t> mInt64Value;
347 Nullable<bool> mBoolValue;
348
349 // std::vector<CustomExtension> handles both Object and Array
350 Nullable<std::vector<CustomExtension>> mValues;
351
352 operator bool() const {
353 return Size() != 0;
354 }
355
356 size_t Size() const {
357 if (mValues.isPresent) {
358 return mValues.value.size();
359 } else if (mStringValue.isPresent || mDoubleValue.isPresent || mUint64Value.isPresent || mInt64Value.isPresent || mBoolValue.isPresent) {
360 return 1;
361 }
362 return 0;
363 }
364
365 CustomExtension() = default;
366
367 ~CustomExtension() = default;
368
369 CustomExtension(const CustomExtension &other) = default;
370
371 CustomExtension& operator=(const CustomExtension&) = default;
372};
