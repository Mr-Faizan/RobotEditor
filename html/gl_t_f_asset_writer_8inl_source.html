63 template<typename T, size_t N>
64 inline
65 Value& MakeValue(Value& val, T(&r)[N], MemoryPoolAllocator<>& al) {
66 val.SetArray();
67 val.Reserve(N, al);
68 for (decltype(N) i = 0; i < N; ++i) {
69 val.PushBack(r[i], al);
70 }
71 return val;
72 }

79 for (unsigned int i = 0; i < r.size(); ++i) {

97 if (v.empty()) return;

99 lst.SetArray();

104 obj.AddMember(StringRef(fieldId), lst, al);

222 inline void Write(Value& obj, Image& img, AssetWriter& w)
223 {
224 std::string uri;
225 if (w.mAsset.extensionsUsed.KHR_binary_glTF && img.bufferView) {
226 Value exts, ext;
227 exts.SetObject();
228 ext.SetObject();
229
230 ext.AddMember("bufferView", StringRef(img.bufferView->id), w.mAl);
231
232 if (!img.mimeType.empty())
233 ext.AddMember("mimeType", StringRef(img.mimeType), w.mAl);
234
235 exts.AddMember("KHR_binary_glTF", ext, w.mAl);
236 obj.AddMember("extensions", exts, w.mAl);
237 return;
238 }
239 else if (img.HasData()) {
240 uri = "data:" + (img.mimeType.empty() ? "application/octet-stream" : img.mimeType);
241 uri += ";base64,";
242 Base64::Encode(img.GetData(), img.GetDataLength(), uri);
243 }
244 else {
245 uri = img.uri;
246 }
247
248 obj.AddMember("uri", Value(uri, w.mAl).Move(), w.mAl);
249 }

294 }

299 inline void Write(Value& obj, Mesh& m, AssetWriter& w)
300 {
301 /********************* Name **********************/
302 obj.AddMember("name", m.name, w.mAl);
303
304 /**************** Mesh extensions ****************/
305 if(m.Extension.size() > 0)
306 {
307 Value json_extensions;
308
309 json_extensions.SetObject();
310#ifdef ASSIMP_IMPORTER_GLTF_USE_OPEN3DGC
311 for(Mesh::SExtension* ptr_ext : m.Extension)
312 {
313 switch(ptr_ext->Type)
314 {
315 case Mesh::SExtension::EType::Compression_Open3DGC:
316 {
317 Value json_comp_data;
318 Mesh::SCompression_Open3DGC* ptr_ext_comp = (Mesh::SCompression_Open3DGC*)ptr_ext;
319
320 // filling object "compressedData"
321 json_comp_data.SetObject();
322 json_comp_data.AddMember("buffer", ptr_ext_comp->Buffer, w.mAl);
323 json_comp_data.AddMember("byteOffset", static_cast<uint64_t>(ptr_ext_comp->Offset), w.mAl);
324 json_comp_data.AddMember("componentType", 5121, w.mAl);
325 json_comp_data.AddMember("type", "SCALAR", w.mAl);
326 json_comp_data.AddMember("count", static_cast<uint64_t>(ptr_ext_comp->Count), w.mAl);
327 if(ptr_ext_comp->Binary)
328 json_comp_data.AddMember("mode", "binary", w.mAl);
329 else
330 json_comp_data.AddMember("mode", "ascii", w.mAl);
331
332 json_comp_data.AddMember("indicesCount", static_cast<uint64_t>(ptr_ext_comp->IndicesCount), w.mAl);
333 json_comp_data.AddMember("verticesCount", static_cast<uint64_t>(ptr_ext_comp->VerticesCount), w.mAl);
334 // filling object "Open3DGC-compression"
335 Value json_o3dgc;
336
337 json_o3dgc.SetObject();
338 json_o3dgc.AddMember("compressedData", json_comp_data, w.mAl);
339 // add member to object "extensions"
340 json_extensions.AddMember("Open3DGC-compression", json_o3dgc, w.mAl);
341 }
342
343 break;
344 default:
345 throw DeadlyImportError("GLTF: Can not write mesh: unknown mesh extension, only Open3DGC is supported.");
346 }// switch(ptr_ext->Type)
347 }// for(Mesh::SExtension* ptr_ext : m.Extension)
348#endif
349
350 // Add extensions to mesh
351 obj.AddMember("extensions", json_extensions, w.mAl);
352 }// if(m.Extension.size() > 0)
353
354 /****************** Primitives *******************/
355 Value primitives;
356 primitives.SetArray();
357 primitives.Reserve(unsigned(m.primitives.size()), w.mAl);
358
359 for (size_t i = 0; i < m.primitives.size(); ++i) {
360 Mesh::Primitive& p = m.primitives[i];
361 Value prim;
362 prim.SetObject();
363 {
364 prim.AddMember("mode", Value(int(p.mode)).Move(), w.mAl);
365
366 if (p.material)
367 prim.AddMember("material", p.material->id, w.mAl);
368
369 if (p.indices)
370 prim.AddMember("indices", Value(p.indices->id, w.mAl).Move(), w.mAl);
371
372 Value attrs;
373 attrs.SetObject();
374 {
375 WriteAttrs(w, attrs, p.attributes.position, "POSITION");
376 WriteAttrs(w, attrs, p.attributes.normal, "NORMAL");
377 WriteAttrs(w, attrs, p.attributes.texcoord, "TEXCOORD", true);
378 WriteAttrs(w, attrs, p.attributes.color, "COLOR");
379 WriteAttrs(w, attrs, p.attributes.joint, "JOINT");
380 WriteAttrs(w, attrs, p.attributes.jointmatrix, "JOINTMATRIX");
381 WriteAttrs(w, attrs, p.attributes.weight, "WEIGHT");
382 }
383 prim.AddMember("attributes", attrs, w.mAl);
384 }
385 primitives.PushBack(prim, w.mAl);
386 }
387
388 obj.AddMember("primitives", primitives, w.mAl);
389 }

524 inline void AssetWriter::WriteFile(const char* path)
525 {
526 std::unique_ptr<IOStream> jsonOutFile(mAsset.OpenFile(path, "wt", true));
527
528 if (jsonOutFile == nullptr) {
529 throw DeadlyExportError("Could not open output file: " + std::string(path));
530 }
531
532 StringBuffer docBuffer;
533
534 PrettyWriter<StringBuffer> writer(docBuffer);
535 if (!mDoc.Accept(writer)) {
536 throw DeadlyExportError("Failed to write scene data!");
537 }
538
539 if (jsonOutFile->Write(docBuffer.GetString(), docBuffer.GetSize(), 1) != 1) {
540 throw DeadlyExportError("Failed to write scene data!");
541 }
542
543 // Write buffer data to separate .bin files
544 for (unsigned int i = 0; i < mAsset.buffers.Size(); ++i) {
545 Ref<Buffer> b = mAsset.buffers.Get(i);
546
547 std::string binPath = b->GetURI();
548
549 std::unique_ptr<IOStream> binOutFile(mAsset.OpenFile(binPath, "wb", true));
550
551 if (binOutFile == nullptr) {
552 throw DeadlyExportError("Could not open output file: " + binPath);
553 }
554
555 if (b->byteLength > 0) {
556 if (binOutFile->Write(b->GetPointer(), b->byteLength, 1) != 1) {
557 throw DeadlyExportError("Failed to write binary file: " + binPath);
558 }
559 }
560 }
561 }

563 inline void AssetWriter::WriteGLBFile(const char* path)
564 {
565 std::unique_ptr<IOStream> outfile(mAsset.OpenFile(path, "wb", true));
566
567 if (outfile == nullptr) {
568 throw DeadlyExportError("Could not open output file: " + std::string(path));
569 }
570
571 // we will write the header later, skip its size
572 outfile->Seek(sizeof(GLB_Header), aiOrigin_SET);
573
574 StringBuffer docBuffer;
575 Writer<StringBuffer> writer(docBuffer);
576 if (!mDoc.Accept(writer)) {
577 throw DeadlyExportError("Failed to write scene data!");
578 }
579
580 if (outfile->Write(docBuffer.GetString(), docBuffer.GetSize(), 1) != 1) {
581 throw DeadlyExportError("Failed to write scene data!");
582 }
583
584 WriteBinaryData(outfile.get(), docBuffer.GetSize());
585 }

587 inline void AssetWriter::WriteBinaryData(IOStream* outfile, size_t sceneLength)
588 {
589 //
590 // write the body data
591 //
592
593 size_t bodyLength = 0;
594 if (Ref<Buffer> b = mAsset.GetBodyBuffer()) {
595 bodyLength = b->byteLength;
596
597 if (bodyLength > 0) {
598 size_t bodyOffset = sizeof(GLB_Header) + sceneLength;
599 bodyOffset = (bodyOffset + 3) & ~3; // Round up to next multiple of 4
600
601 outfile->Seek(bodyOffset, aiOrigin_SET);
602
603 if (outfile->Write(b->GetPointer(), b->byteLength, 1) != 1) {
604 throw DeadlyExportError("Failed to write body data!");
605 }
606 }
607 }
608
609 //
610 // write the header
611 //
612
613 GLB_Header header;
614 memcpy(header.magic, AI_GLB_MAGIC_NUMBER, sizeof(header.magic));
615
616 header.version = 1;
617 AI_SWAP4(header.version);
618
619 header.length = uint32_t(sizeof(header) + sceneLength + bodyLength);
620 AI_SWAP4(header.length);
621
622 header.sceneLength = uint32_t(sceneLength);
623 AI_SWAP4(header.sceneLength);
624
625 header.sceneFormat = SceneFormat_JSON;
626 AI_SWAP4(header.sceneFormat);
627
628 outfile->Seek(0, aiOrigin_SET);
629
630 if (outfile->Write(&header, 1, sizeof(header)) != sizeof(header)) {
631 throw DeadlyExportError("Failed to write the header!");
632 }
633 }

664 template<class T>
665 void AssetWriter::WriteObjects(LazyDict<T>& d)
666 {
667 if (d.mObjs.empty()) return;
668
669 Value* container = &mDoc;
670
671 if (d.mExtId) {
672 Value* exts = FindObject(mDoc, "extensions");
673 if (!exts) {
674 mDoc.AddMember("extensions", Value().SetObject().Move(), mDoc.GetAllocator());
675 exts = FindObject(mDoc, "extensions");
676 }
677
678 if (!(container = FindObject(*exts, d.mExtId))) {
679 exts->AddMember(StringRef(d.mExtId), Value().SetObject().Move(), mDoc.GetAllocator());
680 container = FindObject(*exts, d.mExtId);
681 }
682 }
683
684 Value* dict;
685 if (!(dict = FindObject(*container, d.mDictId))) {
686 container->AddMember(StringRef(d.mDictId), Value().SetObject().Move(), mDoc.GetAllocator());
687 dict = FindObject(*container, d.mDictId);
688 }
689
690 for (size_t i = 0; i < d.mObjs.size(); ++i) {
691 if (d.mObjs[i]->IsSpecial()) continue;
