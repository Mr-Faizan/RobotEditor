586struct Animation {
588 std::string mName;
589
591 std::vector<AnimationChannel> mChannels;
592
594 std::vector<Animation *> mSubAnims;
595
597 ~Animation() {
598 for (std::vector<Animation *>::iterator it = mSubAnims.begin(); it != mSubAnims.end(); ++it) {
599 delete *it;
600 }
601 }
602
604 void CollectChannelsRecursively(std::vector<AnimationChannel> &channels) {
605 channels.insert(channels.end(), mChannels.begin(), mChannels.end());
606
607 for (std::vector<Animation *>::iterator it = mSubAnims.begin(); it != mSubAnims.end(); ++it) {
608 Animation *pAnim = (*it);
609 pAnim->CollectChannelsRecursively(channels);
610 }
611 }
612
614 void CombineSingleChannelAnimations() {
615 CombineSingleChannelAnimationsRecursively(this);
616 }
617
618 void CombineSingleChannelAnimationsRecursively(Animation *pParent) {
619 std::set<std::string> childrenTargets;
620 bool childrenAnimationsHaveDifferentChannels = true;
621
622 for (std::vector<Animation *>::iterator it = pParent->mSubAnims.begin(); it != pParent->mSubAnims.end();) {
623 Animation *anim = *it;
624 // Assign the first animation name to the parent if empty.
625 // This prevents the animation name from being lost when animations are combined
626 if (mName.empty()) {
627 mName = anim->mName;
628 }
629 CombineSingleChannelAnimationsRecursively(anim);
630
631 if (childrenAnimationsHaveDifferentChannels && anim->mChannels.size() == 1 &&
632 childrenTargets.find(anim->mChannels[0].mTarget) == childrenTargets.end()) {
633 childrenTargets.insert(anim->mChannels[0].mTarget);
634 } else {
635 childrenAnimationsHaveDifferentChannels = false;
636 }
637
638 ++it;
639 }
640
641 // We only want to combine animations if they have different channels
642 if (childrenAnimationsHaveDifferentChannels) {
643 for (std::vector<Animation *>::iterator it = pParent->mSubAnims.begin(); it != pParent->mSubAnims.end();) {
644 Animation *anim = *it;
645
646 pParent->mChannels.push_back(anim->mChannels[0]);
647
648 it = pParent->mSubAnims.erase(it);
649
650 delete anim;
651 continue;
652 }
653 }
654 }
655};
