38// Class that supports input of glTF (and some simple USD) files, encodes
39// them with Draco compression, and outputs glTF Draco compressed files.
40//
41// glTF supported extensions:
42// Input and Output:
43// KHR_draco_mesh_compression. http://shortn/_L5tPQqdwWf
44// KHR_materials_unlit. http://shortn/_3eaDLoIGam
45// KHR_texture_transform. http://shortn/_PORWgVTEe8
46//
47// glTF unsupported features:
48// Input and Output:
49// Morph targets. http://shortn/_zE5DLw8a9B
50// Sparse accessors. http://shortn/_h3FwbzQl4f
51// KHR_lights_punctual. http://shortn/_nzGk80wKtK
52// KHR_materials_pbrSpecularGlossiness. http://shortn/_iz0VC6dIKe
53// All vendor extensions.
54class DracoTranscoder {
55 public:
56 struct FileOptions {
57 std::string input_filename; // Must be non-empty.
58 std::string output_filename; // Must be non-empty.
59 std::string output_bin_filename = "";
60 std::string output_resource_directory = "";
61 };
62
63 DracoTranscoder();
64
65 // Creates a DracoTranscoder object. |options| sets the compression options
66 // used in the Encode function.
67 static StatusOr<std::unique_ptr<DracoTranscoder>> Create(
68 const DracoTranscodingOptions &options);
69
70 // Deprecated.
71 // TODO(fgalligan): Remove when function is not being used anymore.
72 static StatusOr<std::unique_ptr<DracoTranscoder>> Create(
73 const DracoCompressionOptions &options);
74
75 // Encodes the input with Draco compression using the compression options
76 // passed in the Create function. The recommended use case is to create a
77 // transcoder once and call Transcode for multiple files.
78 Status Transcode(const FileOptions &file_options);
79
80 private:
81 // Read scene from file.
82 Status ReadScene(const FileOptions &file_options);
83
84 // Write scene to file.
85 Status WriteScene(const FileOptions &file_options);
86
87 // Apply compression settings to the scene.
88 Status CompressScene();
89
90 private:
91 GltfEncoder gltf_encoder_;
92
93 // The scene being transcoded.
94 std::unique_ptr<Scene> scene_;
95
96 // Copy of the transcoding options passed into the Create function.
97 DracoTranscodingOptions transcoding_options_;
98};
