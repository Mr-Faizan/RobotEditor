57template <class T>
58inline bool SetGenericProperty(std::map<unsigned int, T> &list,
59 const char *szName, const T &value) {
60 ai_assert(nullptr != szName);
61 const uint32_t hash = SuperFastHash(szName);
62
63 typename std::map<unsigned int, T>::iterator it = list.find(hash);
64 if (it == list.end()) {
65 list.insert(std::pair<unsigned int, T>(hash, value));
66 return false;
67 }
68 (*it).second = value;
69
70 return true;
71}

88// ------------------------------------------------------------------------------------------------
89// Special version for pointer types - they will be deleted when replaced with another value
90// passing nullptr removes the whole property
91template <class T>
92inline void SetGenericPropertyPtr(std::map<unsigned int, T *> &list,
93 const char *szName, T *value, bool *bWasExisting = nullptr) {
94 ai_assert(nullptr != szName);
95 const uint32_t hash = SuperFastHash(szName);
96
97 typename std::map<unsigned int, T *>::iterator it = list.find(hash);
98 if (it == list.end()) {
99 if (bWasExisting) {
100 *bWasExisting = false;
101 }
102
103 list.insert(std::pair<unsigned int, T *>(hash, value));
104 return;
105 }
106 if ((*it).second != value) {
107 delete (*it).second;
108 (*it).second = value;
109 }
110 if (!value) {
111 list.erase(it);
112 }
113 if (bWasExisting) {
114 *bWasExisting = true;
115 }
116}

129
