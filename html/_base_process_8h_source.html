63class SharedPostProcessInfo {
64public:
65 struct Base {
66 virtual ~Base() = default;
67 };
68
70 template <typename T>
71 struct THeapData : public Base {
72 explicit THeapData(T *in) :
73 data(in) {}
74
75 ~THeapData() {
76 delete data;
77 }
78 T *data;
79 };
80
82 template <typename T>
83 struct TStaticData : public Base {
84 explicit TStaticData(T in) :
85 data(in) {}
86
87 ~TStaticData() = default;
88
89 T data;
90 };
91
92 // some typedefs for cleaner code
93 typedef unsigned int KeyType;
94 typedef std::map<KeyType, Base *> PropertyMap;
95
96public:
98 ~SharedPostProcessInfo() {
99 Clean();
100 }
101
103 void Clean() {
104 // invoke the virtual destructor for all stored properties
105 for (PropertyMap::iterator it = pmap.begin(), end = pmap.end();
106 it != end; ++it) {
107 delete (*it).second;
108 }
109 pmap.clear();
110 }
111
113 template <typename T>
114 void AddProperty(const char *name, T *in) {
115 AddProperty(name, (Base *)new THeapData<T>(in));
116 }
117
119 template <typename T>
120 void AddProperty(const char *name, T in) {
121 AddProperty(name, (Base *)new TStaticData<T>(in));
122 }
123
125 template <typename T>
126 bool GetProperty(const char *name, T *&out) const {
127 THeapData<T> *t = (THeapData<T> *)GetPropertyInternal(name);
128 if (!t) {
129 out = nullptr;
130 return false;
131 }
132 out = t->data;
133 return true;
134 }
135
137 template <typename T>
138 bool GetProperty(const char *name, T &out) const {
139 TStaticData<T> *t = (TStaticData<T> *)GetPropertyInternal(name);
140 if ( nullptr == t) {
141 return false;
142 }
143 out = t->data;
144 return true;
145 }
146
148 void RemoveProperty(const char *name) {
149 SetGenericPropertyPtr<Base>(pmap, name, nullptr );
150 }
151
152private:
153 void AddProperty(const char *name, Base *data) {
154 SetGenericPropertyPtr<Base>(pmap, name, data);
155 }
156
157 Base *GetPropertyInternal(const char *name) const {
158 return GetGenericProperty<Base *>(pmap, name, nullptr );
159 }
160
161private:
163 PropertyMap pmap;
164};
