71 // Returns the data from |accessor| as a vector of |T|.
72 template <typename T>
73 static StatusOr<std::vector<T>> CopyDataAsFloat(
74 const tinygltf::Model &model, const tinygltf::Accessor &accessor) {
75 const int num_components = GetNumComponentsForType(accessor.type);
76 if (num_components != T::dimension) {
77 return Status(Status::DRACO_ERROR,
78 "Dimension does not equal num components.");
79 }
80 return CopyDataAsFloatImpl<T>(model, accessor);
81 }
82
83 private:
84 template <typename T>
85 static StatusOr<std::vector<T>> CopyDataAsFloatImpl(
86 const tinygltf::Model &model, const tinygltf::Accessor &accessor) {
87 if (accessor.componentType != TINYGLTF_COMPONENT_TYPE_FLOAT) {
88 return Status(Status::DRACO_ERROR,
89 "Non-float data is not supported by CopyDataAsFloat().");
90 }
91 if (accessor.bufferView < 0) {
92 return Status(Status::DRACO_ERROR,
93 "Error CopyDataAsFloat() bufferView < 0.");
94 }
95
96 const tinygltf::BufferView &buffer_view =
97 model.bufferViews[accessor.bufferView];
98 if (buffer_view.buffer < 0) {
99 return Status(Status::DRACO_ERROR, "Error CopyDataAsFloat() buffer < 0.");
100 }
101
102 const tinygltf::Buffer &buffer = model.buffers[buffer_view.buffer];
103
104 const unsigned char *const data_start =
105 buffer.data.data() + buffer_view.byteOffset + accessor.byteOffset;
106 const int byte_stride = accessor.ByteStride(buffer_view);
107 const int component_size =
108 tinygltf::GetComponentSizeInBytes(accessor.componentType);
109
110 std::vector<T> output;
111 output.resize(accessor.count);
112
113 const int num_components = GetNumComponentsForType(accessor.type);
114 const unsigned char *data = data_start;
115 for (int i = 0; i < accessor.count; ++i) {
116 T values;
117
118 for (int c = 0; c < num_components; ++c) {
119 float value = 0.0f;
120 memcpy(&value, data + (c * component_size), component_size);
121 SetDataImpl<T>(value, c, &values);
122 }
123
124 output[i] = values;
125 data += byte_stride;
126 }
127
128 return output;
129 }
130
131 template <typename T>
132 static void SetDataImpl(float value, int index, T *values) {
133 (*values)[index] = value;
134 }
