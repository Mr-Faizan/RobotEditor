114 f(file) {}

116 fclose(f);

118
119 size_t Read(void *b, size_t sz, size_t n) { return fread(b, sz, n, f); }
120 size_t Write(const void *b, size_t sz, size_t n) { return fwrite(b, sz, n, f); }
121 int Seek(size_t off, aiOrigin orig) { return fseek(f, off, int(orig)); }
122 size_t Tell() const { return ftell(f); }
123
124 size_t FileSize() {
125 long p = Tell(), len = (Seek(0, aiOrigin_END), Tell());

281template <class T>
282struct ReadHelper {
283 static bool Read(Value &val, T &out) {
284 return val.IsInt() ? out = static_cast<T>(val.GetInt()), true : false;
285 }
286};
287
288template <>
289struct ReadHelper<bool> {
290 static bool Read(Value &val, bool &out) {
291 return val.IsBool() ? out = val.GetBool(), true : false;
292 }
293};
294
295template <>
296struct ReadHelper<float> {
297 static bool Read(Value &val, float &out) {
298 return val.IsNumber() ? out = static_cast<float>(val.GetDouble()), true : false;
299 }
300};
301
302template <unsigned int N>
303struct ReadHelper<float[N]> {
304 static bool Read(Value &val, float (&out)[N]) {
305 if (!val.IsArray() || val.Size() != N) return false;
306 for (unsigned int i = 0; i < N; ++i) {
307 if (val[i].IsNumber())
308 out[i] = static_cast<float>(val[i].GetDouble());
309 }
310 return true;
311 }
312};
313
314template <>
315struct ReadHelper<const char *> {
316 static bool Read(Value &val, const char *&out) {
317 return val.IsString() ? (out = val.GetString(), true) : false;
318 }
319};
320
321template <>
322struct ReadHelper<std::string> {
323 static bool Read(Value &val, std::string &out) {
324 return val.IsString() ? (out = std::string(val.GetString(), val.GetStringLength()), true) : false;
325 }
326};

328template <class T>
329struct ReadHelper<Nullable<T>> {
330 static bool Read(Value &val, Nullable<T> &out) {
331 return out.isPresent = ReadHelper<T>::Read(val, out.value);

333};

354template <class T>
355inline static bool ReadMember(Value &obj, const char *id, T &out) {
356 if (!obj.IsObject()) {
357 return false;
358 }
359 Value::MemberIterator it = obj.FindMember(id);
360 if (it != obj.MemberEnd()) {
361 return ReadHelper<T>::Read(it->value, out);
362 }
363 return false;
364}

366template <class T>
367inline static T MemberOrDefault(Value &obj, const char *id, T defaultValue) {
368 T out;
369 return ReadMember(obj, id, out) ? out : defaultValue;
370}

380template <int N>
381inline void throwUnexpectedTypeError(const char (&expectedTypeName)[N], const char *memberId, const char *context, const char *extraContext) {
382 std::string fullContext = context;
383 if (extraContext && (strlen(extraContext) > 0)) {
384 fullContext = fullContext + " (" + extraContext + ")";
385 }
386 throw DeadlyImportError("Member \"", memberId, "\" was not of type \"", expectedTypeName, "\" when reading ", fullContext);
387}
