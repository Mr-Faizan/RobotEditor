23// Decoder for attribute values encoded with the standard parallelogram
24// prediction. See the description of the corresponding encoder for more
25// details.
26template <typename DataTypeT, class TransformT, class MeshDataT>
27class MeshPredictionSchemeParallelogramDecoder
28 : public MeshPredictionSchemeDecoder<DataTypeT, TransformT, MeshDataT> {
29 public:
30 using CorrType =
31 typename PredictionSchemeDecoder<DataTypeT, TransformT>::CorrType;
32 using CornerTable = typename MeshDataT::CornerTable;
33 explicit MeshPredictionSchemeParallelogramDecoder(
34 const PointAttribute *attribute)
35 : MeshPredictionSchemeDecoder<DataTypeT, TransformT, MeshDataT>(
36 attribute) {}
37 MeshPredictionSchemeParallelogramDecoder(const PointAttribute *attribute,
38 const TransformT &transform,
39 const MeshDataT &mesh_data)
40 : MeshPredictionSchemeDecoder<DataTypeT, TransformT, MeshDataT>(
41 attribute, transform, mesh_data) {}
42
43 bool ComputeOriginalValues(const CorrType *in_corr, DataTypeT *out_data,
44 int size, int num_components,
45 const PointIndex *entry_to_point_id_map) override;
46 PredictionSchemeMethod GetPredictionMethod() const override {
47 return MESH_PREDICTION_PARALLELOGRAM;
48 }
49
50 bool IsInitialized() const override {
51 return this->mesh_data().IsInitialized();
52 }
53};
54
55template <typename DataTypeT, class TransformT, class MeshDataT>
56bool MeshPredictionSchemeParallelogramDecoder<DataTypeT, TransformT,
57 MeshDataT>::
58 ComputeOriginalValues(const CorrType *in_corr, DataTypeT *out_data,
59 int /* size */, int num_components,
60 const PointIndex * /* entry_to_point_id_map */) {
61 this->transform().Init(num_components);
62
63 const CornerTable *const table = this->mesh_data().corner_table();
64 const std::vector<int32_t> *const vertex_to_data_map =
65 this->mesh_data().vertex_to_data_map();
66
67 // For storage of prediction values (already initialized to zero).
68 std::unique_ptr<DataTypeT[]> pred_vals(new DataTypeT[num_components]());
69
70 // Restore the first value.
71 this->transform().ComputeOriginalValue(pred_vals.get(), in_corr, out_data);
72
73 const int corner_map_size =
74 static_cast<int>(this->mesh_data().data_to_corner_map()->size());
75 for (int p = 1; p < corner_map_size; ++p) {
76 const CornerIndex corner_id = this->mesh_data().data_to_corner_map()->at(p);
77 const int dst_offset = p * num_components;
78 if (!ComputeParallelogramPrediction(p, corner_id, table,
79 *vertex_to_data_map, out_data,
80 num_components, pred_vals.get())) {
81 // Parallelogram could not be computed, Possible because some of the
82 // vertices are not valid (not encoded yet).
83 // We use the last encoded point as a reference (delta coding).
84 const int src_offset = (p - 1) * num_components;
85 this->transform().ComputeOriginalValue(
86 out_data + src_offset, in_corr + dst_offset, out_data + dst_offset);
87 } else {
88 // Apply the parallelogram prediction.
89 this->transform().ComputeOriginalValue(
90 pred_vals.get(), in_corr + dst_offset, out_data + dst_offset);
91 }
92 }
93 return true;
94}
