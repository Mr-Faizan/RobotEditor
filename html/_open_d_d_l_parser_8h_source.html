48template <class T>
49inline T *lookForNextToken(T *in, T *end) {
50 while ((in != end) && (isSpace(*in) || isNewLine(*in) || ',' == *in)) {
51 ++in;
52 }
53 return in;
54}

61template <class T>
62inline T *getNextToken(T *in, T *end) {
63 T *tmp(in);
64 in = lookForNextToken(in, end);
65 if (tmp == in) {
66 ++in;
67 }
68 return in;
69}
