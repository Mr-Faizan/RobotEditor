67 // std::plus-family operates on operands with identical types - we need to
68 // support all the (vectype op float) combinations in vector maths.
69 // Providing T(float) would open the way to endless implicit conversions.
71 namespace Intern {
72 template <typename T0, typename T1, typename TRES = T0> struct plus {
73 TRES operator() (const T0& t0, const T1& t1) const {
74 return t0+t1;
75 }
76 };
77 template <typename T0, typename T1, typename TRES = T0> struct minus {
78 TRES operator() (const T0& t0, const T1& t1) const {
79 return t0-t1;
80 }
81 };
82 template <typename T0, typename T1, typename TRES = T0> struct multiplies {
83 TRES operator() (const T0& t0, const T1& t1) const {
84 return t0*t1;
85 }
86 };
87 template <typename T0, typename T1, typename TRES = T0> struct divides {
88 TRES operator() (const T0& t0, const T1& t1) const {
89 return t0/t1;
90 }
91 };
92 }

121

125
126 if (msh->HasTangentsAndBitangents()) {

130

132 texcoords[i] = msh->mTextureCoords[i][idx];

214 template <template <typename t> class op> static Vertex BinaryOp(const Vertex& v0, const Vertex& v1) {
215 // this is a heavy task for the compiler to optimize ... *pray*
216
217 Vertex res;
218 res.position = op<aiVector3D>()(v0.position,v1.position);
219 res.normal = op<aiVector3D>()(v0.normal,v1.normal);
220 res.tangent = op<aiVector3D>()(v0.tangent,v1.tangent);
221 res.bitangent = op<aiVector3D>()(v0.bitangent,v1.bitangent);
222
223 for (unsigned int i = 0; i < AI_MAX_NUMBER_OF_TEXTURECOORDS; ++i) {
224 res.texcoords[i] = op<aiVector3D>()(v0.texcoords[i],v1.texcoords[i]);
225 }
226 for (unsigned int i = 0; i < AI_MAX_NUMBER_OF_COLOR_SETS; ++i) {
227 res.colors[i] = op<aiColor4D>()(v0.colors[i],v1.colors[i]);
228 }
229 return res;
230 }
