74// Class used to keep track of the json state.
75class JsonWriter {
76 public:
77 enum OutputType { START, BEGIN, END, VALUE };
78 enum Mode { READABLE, COMPACT };
79
80 JsonWriter()
81 : last_type_(START), mode_(READABLE), indent_(*this), separator_(*this) {}
82 void SetMode(Mode mode) { mode_ = mode; }
83
84 // Clear the stringstream and set last type to START.
85 void Reset();
86
87 // Every call to BeginObject should have a matching call to EndObject.
88 void BeginObject();
89 void BeginObject(const std::string &name);
90 void EndObject();
91
92 // Every call to BeginArray should have a matching call to EndArray.
93 void BeginArray(const std::string &name);
94 void EndArray();
95
96 template <typename T>
97 void OutputValue(const T &value) {
98 FinishPreviousLine(VALUE);
99 o_ << indent_ << std::setprecision(17) << value;
100 }
101
102 void OutputValue(const bool &value) {
103 FinishPreviousLine(VALUE);
104 o_ << indent_ << ToString(value);
105 }
106
107 void OutputValue(const std::string &name) {
108 const std::string escaped_name = EscapeJsonSpecialCharacters(name);
109 FinishPreviousLine(VALUE);
110 o_ << indent_ << "\"" << escaped_name << "\"";
111 }
112
113 void OutputValue(const std::string &name, const std::string &value) {
114 const std::string escaped_name = EscapeJsonSpecialCharacters(name);
115 const std::string escaped_value = EscapeJsonSpecialCharacters(value);
116 FinishPreviousLine(VALUE);
117 o_ << indent_ << "\"" << escaped_name << "\":" << separator_ << "\""
118 << escaped_value << "\"";
119 }
