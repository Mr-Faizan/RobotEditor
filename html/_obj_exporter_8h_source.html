154 template <class T, class Compare = std::less<T>>
155 class indexMap {
156 int mNextIndex;
157 typedef std::map<T, int, Compare> dataType;
158 dataType vecMap;
159
160 public:
161 indexMap()
162 : mNextIndex(1) {
163 // empty
164 }
165
166 int getIndex(const T& key) {
167 typename dataType::iterator vertIt = vecMap.find(key);
168 // vertex already exists, so reference it
169 if(vertIt != vecMap.end()){
170 return vertIt->second;
171 }
172 return vecMap[key] = mNextIndex++;
173 };
174
175 void getKeys( std::vector<T>& keys ) {
176 keys.resize(vecMap.size());
177 for(typename dataType::iterator it = vecMap.begin(); it != vecMap.end(); ++it){
178 keys[it->second-1] = it->first;
179 }
180 };
181 };
