35// Class used to hold all of the geometry to create a scene. A scene is
36// comprised of one or more meshes, one or more scene nodes, one or more
37// mesh groups, and a material library. The meshes are defined in their
38// local space. A mesh group is a list of meshes. The scene nodes create
39// a scene hierarchy to transform meshes in their local space into scene space.
40// The material library contains all of the materials and textures used by the
41// meshes in this scene.
42class Scene {
43 public:
44 Scene() {}
45
46 void Copy(const Scene &s);
47
48 // Adds a Draco |mesh| to the scene. Returns the index to the stored mesh or
49 // |kInvalidMeshIndex| if the mesh is a nullptr.
50 MeshIndex AddMesh(std::unique_ptr<Mesh> mesh) {
51 if (mesh == nullptr) {
52 return kInvalidMeshIndex;
53 }
54 meshes_.push_back(std::move(mesh));
55 return MeshIndex(meshes_.size() - 1);
56 }
57
58 // Removes base mesh and corresponding material at |index|, removes references
59 // to removed base mesh and corresponding materials from mesh groups, and
60 // updates references to remaining base meshes in mesh groups.
61 Status RemoveMesh(MeshIndex index);
62
63 // Returns the number of meshes in a scene before instancing is applied.
64 int NumMeshes() const { return meshes_.size(); }
65
66 // Returns a mesh in the scene before instancing is applied. The mesh
67 // coordinates are local to the mesh.
68 Mesh &GetMesh(MeshIndex index) { return *meshes_[index]; }
69 const Mesh &GetMesh(MeshIndex index) const { return *meshes_[index]; }
70
71 // Creates a mesh group and returns the index to the mesh group.
72 MeshGroupIndex AddMeshGroup() {
73 std::unique_ptr<MeshGroup> mesh(new MeshGroup());
74 mesh_groups_.push_back(std::move(mesh));
75 return MeshGroupIndex(mesh_groups_.size() - 1);
76 }
77
78 // Removes mesh group at |index|, invalidates references to removed mesh group
79 // in scene nodes, and updates references to remaining mesh groups in scene
80 // nodes.
81 Status RemoveMeshGroup(MeshGroupIndex index);
82
83 // Removes unused material at |index| and updates references to materials at
84 // indices greater than |index|. Returns error status when |index| is out of
85 // valid range and when material at |index| is used in the scene.
86 Status RemoveMaterial(int index);
87
88 // Returns the number of mesh groups in a scene.
89 int NumMeshGroups() const { return mesh_groups_.size(); }
90
91 // Returns a mesh group in the scene.
92 MeshGroup *GetMeshGroup(MeshGroupIndex index) {
93 return mesh_groups_[index].get();
94 }
95 const MeshGroup *GetMeshGroup(MeshGroupIndex index) const {
96 return mesh_groups_[index].get();
97 }
98
99 // Creates a scene node and returns the index to the node.
100 SceneNodeIndex AddNode() {
101 std::unique_ptr<SceneNode> node(new SceneNode());
102 nodes_.push_back(std::move(node));
103 return SceneNodeIndex(nodes_.size() - 1);
104 }
105
106 // Returns the number of nodes in a scene.
107 int NumNodes() const { return nodes_.size(); }
108
109 // Returns a node in the scene.
110 SceneNode *GetNode(SceneNodeIndex index) { return nodes_[index].get(); }
111 const SceneNode *GetNode(SceneNodeIndex index) const {
112 return nodes_[index].get();
113 }
114
115 // Either allocates new nodes or removes existing nodes that are beyond
116 // |num_nodes|.
117 void ResizeNodes(int num_nodes) {
118 const size_t old_num_nodes = nodes_.size();
119 nodes_.resize(num_nodes);
120 for (SceneNodeIndex i(old_num_nodes); i < num_nodes; ++i) {
121 nodes_[i].reset(new SceneNode());
122 }
123 }
124
125 // Returns the number of root node indices in a scene.
126 int NumRootNodes() const { return root_node_indices_.size(); }

129 return root_node_indices_;

132 root_node_indices_.push_back(index);
133 }
134 void SetRootNodeIndex(int i, SceneNodeIndex index) {
135 root_node_indices_[i] = index;
136 }
137 void RemoveAllRootNodeIndices() { root_node_indices_.clear(); }

224 private:
225 IndexTypeVector<MeshIndex, std::unique_ptr<Mesh>> meshes_;
226 IndexTypeVector<MeshGroupIndex, std::unique_ptr<MeshGroup>> mesh_groups_;
227 IndexTypeVector<SceneNodeIndex, std::unique_ptr<SceneNode>> nodes_;
228 std::vector<SceneNodeIndex> root_node_indices_;
229 IndexTypeVector<AnimationIndex, std::unique_ptr<Animation>> animations_;
230 IndexTypeVector<SkinIndex, std::unique_ptr<Skin>> skins_;
231
232 // The lights will be written to the output scene but not used for internal
233 // rendering in Draco, e.g, while computing distortion metric.
234 IndexTypeVector<LightIndex, std::unique_ptr<Light>> lights_;
235
236 // The mesh group instance array information will be written to the output
237 // scene but not processed by Draco simplifier modules.
238 IndexTypeVector<InstanceArrayIndex, std::unique_ptr<InstanceArray>>
239 instance_arrays_;
240
241 // Materials used by this scene.
242 MaterialLibrary material_library_;
243
244 // Texture library for storing non-material textures used by this scene, e.g.,
245 // textures containing mesh feature IDs of EXT_mesh_features glTF extension.
246 // Note that scene meshes contain pointers to non-material textures. It is
247 // responsibility of class user to update these pointers when updating the
248 // textures. See Scene::Copy() for example.
249 TextureLibrary non_material_texture_library_;
250
251 // Structural metadata defined by the EXT_structural_metadata glTF extension.
252 StructuralMetadata structural_metadata_;
253};
