29// The main class of this file is PointDVector providing an interface similar to
30// std::vector<PointD> for arbitrary number of dimensions (without a template
31// argument). PointDVectorIterator is a random access iterator, which allows for
32// compatibility with existing algorithms. PseudoPointD provides for a view on
33// the individual items in a contiguous block of memory, which is compatible
34// with the swap function and is returned by a dereference of
35// PointDVectorIterator. Swap functions provide for compatibility/specialization
36// that allows these classes to work with currently utilized STL functions.
37
38// This class allows for swap functionality from the RandomIterator
39// It seems problematic to bring this inside PointDVector due to templating.
40template <typename internal_t>
41class PseudoPointD {
42 public:
43 PseudoPointD(internal_t *mem, internal_t dimension)
44 : mem_(mem), dimension_(dimension) {}
45
46 // Specifically copies referenced memory
47 void swap(PseudoPointD &other) noexcept {
48 for (internal_t dim = 0; dim < dimension_; dim += 1) {
49 std::swap(mem_[dim], other.mem_[dim]);
50 }
51 }
52
53 PseudoPointD(const PseudoPointD &other)
54 : mem_(other.mem_), dimension_(other.dimension_) {}
55
56 const internal_t &operator[](const size_t &n) const {
57 DRACO_DCHECK_LT(n, dimension_);
58 return mem_[n];
59 }
60 internal_t &operator[](const size_t &n) {
61 DRACO_DCHECK_LT(n, dimension_);
62 return mem_[n];
63 }
64
65 bool operator==(const PseudoPointD &other) const {
66 for (auto dim = 0; dim < dimension_; dim += 1) {
67 if (mem_[dim] != other.mem_[dim]) {
68 return false;
69 }
70 }
71 return true;
72 }
73 bool operator!=(const PseudoPointD &other) const {
74 return !this->operator==(other);
75 }
76
77 private:
78 internal_t *const mem_;
79 const internal_t dimension_;
80};

94template <typename internal_t>
95class PointDVector {
96 public:
97 PointDVector(const uint32_t n_items, const uint32_t dimensionality)
98 : n_items_(n_items),
99 dimensionality_(dimensionality),
100 item_size_bytes_(dimensionality * sizeof(internal_t)),
101 data_(n_items * dimensionality),
102 data0_(data_.data()) {}

215 // copy a single contiguous 'item' from one PointDVector into this one.
216 void CopyItem(const PointDVector &source, const internal_t source_index,
217 const internal_t destination_index) {
218 DRACO_DCHECK(&source != this ||
219 (&source == this && source_index != destination_index));
220 DRACO_DCHECK_LT(destination_index, n_items_);
221 DRACO_DCHECK_LT(source_index, source.n_items_);
222
223 // DRACO_DCHECK_EQ(source.n_items_, n_items_); // not technically necessary
224 DRACO_DCHECK_EQ(source.dimensionality_, dimensionality_);
225
226 const internal_t *ref = source[source_index];
227 internal_t *const dest = this->operator[](destination_index);
228 std::memcpy(dest, ref, item_size_bytes_);
229 }

231 // Copy data directly off of an attribute buffer interleaved into internal
232 // memory.
233 void CopyAttribute(
234 // The dimensionality of the attribute being integrated
235 const internal_t attribute_dimensionality,
236 // The offset in dimensions to insert this attribute.
237 const internal_t offset_dimensionality, const internal_t index,
238 // The direct pointer to the data
239 const void *const attribute_item_data) {
240 // chunk copy
241 const size_t copy_size = sizeof(internal_t) * attribute_dimensionality;
242
243 // a multiply and add can be optimized away with an iterator
244 std::memcpy(data0_ + index * dimensionality_ + offset_dimensionality,
245 attribute_item_data, copy_size);
246 }
247 // Copy data off of a contiguous buffer interleaved into internal memory
248 void CopyAttribute(
249 // The dimensionality of the attribute being integrated
250 const internal_t attribute_dimensionality,
251 // The offset in dimensions to insert this attribute.
252 const internal_t offset_dimensionality,
253 const internal_t *const attribute_mem) {
254 DRACO_DCHECK_LT(offset_dimensionality,
255 dimensionality_ - attribute_dimensionality);
256 // degenerate case block copy the whole buffer.
257 if (dimensionality_ == attribute_dimensionality) {
258 DRACO_DCHECK_EQ(offset_dimensionality, 0);
259 const size_t copy_size =
260 sizeof(internal_t) * attribute_dimensionality * n_items_;
261 std::memcpy(data0_, attribute_mem, copy_size);
262 } else { // chunk copy
263 const size_t copy_size = sizeof(internal_t) * attribute_dimensionality;
264 internal_t *internal_data;
265 const internal_t *attribute_data;
266 internal_t item;
267 for (internal_data = data0_ + offset_dimensionality,
268 attribute_data = attribute_mem, item = 0;
269 item < n_items_; internal_data += dimensionality_,
270 attribute_data += attribute_dimensionality, item += 1) {
271 std::memcpy(internal_data, attribute_data, copy_size);
272 }
273 }
274 }
275
276 private:
277 // internal parameters.
278 const uint32_t n_items_;
279 const uint32_t dimensionality_; // The dimension of the points in the buffer
280 const uint32_t item_size_bytes_;
281 std::vector<internal_t> data_; // contiguously stored data. Never resized.
282 internal_t *const data0_; // raw pointer to base data.
283};
