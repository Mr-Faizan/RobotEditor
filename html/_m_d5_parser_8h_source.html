244// Convert a quaternion to its usual representation
245inline void ConvertQuaternion (const aiVector3D& in, aiQuaternion& out) {
246
247 out.x = in.x;
248 out.y = in.y;
249 out.z = in.z;
250
251 const float t = 1.0f - (in.x*in.x) - (in.y*in.y) - (in.z*in.z);
252
253 if (t < 0.0f) {
254 out.w = 0.0f;
255 } else {
256 out.w = std::sqrt (t);
257 }
258
259 // Assimp convention.
260 out.w *= -1.f;
261}

419// -------------------------------------------------------------------
420inline bool MD5Parser::SkipSpacesAndLineEnd( const char* in, const char** out) {
421 if (in == bufferEnd) {
422 *out = in;
423 return false;
424 }
425
426 bool bHad = false, running = true;
427 while (running) {
428 if( *in == '\r' || *in == '\n') {
429 // we open files in binary mode, so there could be \r\n sequences ...
430 if (!bHad) {
431 bHad = true;
432 ++lineNumber;
433 }
434 } else if (*in == '\t' || *in == ' ') {
435 bHad = false;
436 } else {
437 break;
438 }
439 ++in;
440 if (in == bufferEnd) {
441 break;
442 }
443 }
444 *out = in;
445 return *in != '\0';
446}
