109 inline void Write(Value& obj, Accessor& a, AssetWriter& w)
110 {
111 if (a.bufferView) {
112 obj.AddMember("bufferView", a.bufferView->index, w.mAl);
113 obj.AddMember("byteOffset", (unsigned int)a.byteOffset, w.mAl);
114 }
115 obj.AddMember("componentType", int(a.componentType), w.mAl);
116 obj.AddMember("count", (unsigned int)a.count, w.mAl);
117 obj.AddMember("type", StringRef(AttribType::ToString(a.type)), w.mAl);
118 Value vTmpMax, vTmpMin;
119 if (a.componentType == ComponentType_FLOAT) {
120 obj.AddMember("max", MakeValue(vTmpMax, a.max, w.mAl), w.mAl);
121 obj.AddMember("min", MakeValue(vTmpMin, a.min, w.mAl), w.mAl);
122 } else {
123 obj.AddMember("max", MakeValueCast<int64_t>(vTmpMax, a.max, w.mAl), w.mAl);
124 obj.AddMember("min", MakeValueCast<int64_t>(vTmpMin, a.min, w.mAl), w.mAl);
125 }
126
127 if (a.sparse) {
128 Value sparseValue;
129 sparseValue.SetObject();
130
131 //count
132 sparseValue.AddMember("count", (unsigned int)a.sparse->count, w.mAl);
133
134 //indices
135 Value indices;
136 indices.SetObject();
137 indices.AddMember("bufferView", a.sparse->indices->index, w.mAl);
138 indices.AddMember("byteOffset", (unsigned int)a.sparse->indicesByteOffset, w.mAl);
139 indices.AddMember("componentType", int(a.sparse->indicesType), w.mAl);
140 sparseValue.AddMember("indices", indices, w.mAl);
141
142 //values

144 values.SetObject();
145 values.AddMember("bufferView", a.sparse->values->index, w.mAl);
146 values.AddMember("byteOffset", (unsigned int)a.sparse->valuesByteOffset, w.mAl);

149 obj.AddMember("sparse", sparseValue, w.mAl);
150 }

153 inline void Write(Value& obj, Animation& a, AssetWriter& w)
154 {
155 /****************** Channels *******************/
156 Value channels;
157 channels.SetArray();
158 channels.Reserve(unsigned(a.channels.size()), w.mAl);
159
160 for (size_t i = 0; i < unsigned(a.channels.size()); ++i) {
161 Animation::Channel& c = a.channels[i];
162 Value valChannel;
163 valChannel.SetObject();
164 {
165 valChannel.AddMember("sampler", c.sampler, w.mAl);
166
167 Value valTarget;
168 valTarget.SetObject();
169 {
170 valTarget.AddMember("node", c.target.node->index, w.mAl);
171 switch (c.target.path) {
172 case AnimationPath_TRANSLATION:
173 valTarget.AddMember("path", "translation", w.mAl);
174 break;
175 case AnimationPath_ROTATION:
176 valTarget.AddMember("path", "rotation", w.mAl);
177 break;
178 case AnimationPath_SCALE:
179 valTarget.AddMember("path", "scale", w.mAl);
180 break;
181 case AnimationPath_WEIGHTS:
182 valTarget.AddMember("path", "weights", w.mAl);
183 break;
184 }
185 }
186 valChannel.AddMember("target", valTarget, w.mAl);
187 }
188 channels.PushBack(valChannel, w.mAl);
189 }
190 obj.AddMember("channels", channels, w.mAl);
191
192 /****************** Samplers *******************/
193 Value valSamplers;
194 valSamplers.SetArray();
195
196 for (size_t i = 0; i < unsigned(a.samplers.size()); ++i) {
197 Animation::Sampler& s = a.samplers[i];
198 Value valSampler;
199 valSampler.SetObject();
200 {
201 valSampler.AddMember("input", s.input->index, w.mAl);
202 switch (s.interpolation) {
203 case Interpolation_LINEAR:
204 valSampler.AddMember("interpolation", "LINEAR", w.mAl);
205 break;
206 case Interpolation_STEP:
207 valSampler.AddMember("interpolation", "STEP", w.mAl);
208 break;
209 case Interpolation_CUBICSPLINE:
210 valSampler.AddMember("interpolation", "CUBICSPLINE", w.mAl);
211 break;
212 }
213 valSampler.AddMember("output", s.output->index, w.mAl);
214 }
215 valSamplers.PushBack(valSampler, w.mAl);
216 }
217 obj.AddMember("samplers", valSamplers, w.mAl);
218 }

572 inline void Write(Value& obj, Mesh& m, AssetWriter& w)
573 {
574 /****************** Primitives *******************/
575 Value primitives;
576 primitives.SetArray();
577 primitives.Reserve(unsigned(m.primitives.size()), w.mAl);
578
579 for (size_t i = 0; i < m.primitives.size(); ++i) {
580 Mesh::Primitive& p = m.primitives[i];
581 Value prim;
582 prim.SetObject();
583
584 // Extensions
585 if (p.ngonEncoded)
586 {
587 Value exts;
588 exts.SetObject();
589
590 Value FB_ngon_encoding;
591 FB_ngon_encoding.SetObject();
592
593 exts.AddMember(StringRef("FB_ngon_encoding"), FB_ngon_encoding, w.mAl);
594 prim.AddMember("extensions", exts, w.mAl);
595 }
596
597 {
598 prim.AddMember("mode", Value(int(p.mode)).Move(), w.mAl);
599
600 if (p.material)
601 prim.AddMember("material", p.material->index, w.mAl);
602
603 if (p.indices)
604 prim.AddMember("indices", p.indices->index, w.mAl);
605
606 Value attrs;
607 attrs.SetObject();
608 {
609 WriteAttrs(w, attrs, p.attributes.position, "POSITION");
610 WriteAttrs(w, attrs, p.attributes.normal, "NORMAL");
611 WriteAttrs(w, attrs, p.attributes.texcoord, "TEXCOORD", true);
612 WriteAttrs(w, attrs, p.attributes.color, "COLOR", true);
613 WriteAttrs(w, attrs, p.attributes.joint, "JOINTS", true);
614 WriteAttrs(w, attrs, p.attributes.weight, "WEIGHTS", true);
615 }
616 prim.AddMember("attributes", attrs, w.mAl);
617
618 // targets for blendshapes
619 if (p.targets.size() > 0) {
620 Value tjs;
621 tjs.SetArray();
622 tjs.Reserve(unsigned(p.targets.size()), w.mAl);
623 for (unsigned int t = 0; t < p.targets.size(); ++t) {
624 Value tj;
625 tj.SetObject();
626 {
627 WriteAttrs(w, tj, p.targets[t].position, "POSITION");
628 WriteAttrs(w, tj, p.targets[t].normal, "NORMAL");
629 WriteAttrs(w, tj, p.targets[t].tangent, "TANGENT");
630 }
631 tjs.PushBack(tj, w.mAl);
632 }
633 prim.AddMember("targets", tjs, w.mAl);
634 }
635 }
636 primitives.PushBack(prim, w.mAl);
637 }
638
639 obj.AddMember("primitives", primitives, w.mAl);
640 // targetNames
641 if (m.targetNames.size() > 0) {
642 Value extras;
643 extras.SetObject();
644 Value targetNames;
645 targetNames.SetArray();
646 targetNames.Reserve(unsigned(m.targetNames.size()), w.mAl);
647 for (unsigned int n = 0; n < m.targetNames.size(); ++n) {
648 std::string name = m.targetNames[n];
649 Value tname;
650 tname.SetString(name.c_str(), w.mAl);
651 targetNames.PushBack(tname, w.mAl);
652 }
653 extras.AddMember("targetNames", targetNames, w.mAl);
654 obj.AddMember("extras", extras, w.mAl);
655 }

658 inline void WriteExtrasValue(Value &parent, const CustomExtension &value, AssetWriter &w) {
659 Value valueNode;
660

870 inline void AssetWriter::WriteGLBFile(const char* path)
871 {
872 std::unique_ptr<IOStream> outfile(mAsset.OpenFile(path, "wb", true));
873
874 if (outfile == nullptr) {
875 throw DeadlyExportError("Could not open output file: " + std::string(path));
876 }
877
878 Ref<Buffer> bodyBuffer = mAsset.GetBodyBuffer();
879 if (bodyBuffer->byteLength > 0) {
880 rapidjson::Value glbBodyBuffer;
881 glbBodyBuffer.SetObject();
882 glbBodyBuffer.AddMember("byteLength", static_cast<uint64_t>(bodyBuffer->byteLength), mAl);
883 mDoc["buffers"].PushBack(glbBodyBuffer, mAl);
884 }
885
886 // Padding with spaces as required by the spec
887 uint32_t padding = 0x20202020;
888
889 //
890 // JSON chunk
891 //
892
893 StringBuffer docBuffer;
894 Writer<StringBuffer> writer(docBuffer);
895 if (!mDoc.Accept(writer)) {
896 throw DeadlyExportError("Failed to write scene data!");
897 }
898
899 uint32_t jsonChunkLength = static_cast<uint32_t>((docBuffer.GetSize() + 3) & ~3); // Round up to next multiple of 4
900 auto paddingLength = jsonChunkLength - docBuffer.GetSize();
901
902 GLB_Chunk jsonChunk;
903 jsonChunk.chunkLength = jsonChunkLength;
904 jsonChunk.chunkType = ChunkType_JSON;
905 AI_SWAP4(jsonChunk.chunkLength);
906
907 outfile->Seek(sizeof(GLB_Header), aiOrigin_SET);
908 if (outfile->Write(&jsonChunk, 1, sizeof(GLB_Chunk)) != sizeof(GLB_Chunk)) {
909 throw DeadlyExportError("Failed to write scene data header!");
910 }
911 if (outfile->Write(docBuffer.GetString(), 1, docBuffer.GetSize()) != docBuffer.GetSize()) {
912 throw DeadlyExportError("Failed to write scene data!");
913 }
914 if (paddingLength && outfile->Write(&padding, 1, paddingLength) != paddingLength) {
915 throw DeadlyExportError("Failed to write scene data padding!");
916 }
917
918 //
919 // Binary chunk
920 //
921
922 int GLB_Chunk_count = 1;
923 uint32_t binaryChunkLength = 0;
924 if (bodyBuffer->byteLength > 0) {
925 binaryChunkLength = static_cast<uint32_t>((bodyBuffer->byteLength + 3) & ~3); // Round up to next multiple of 4
926
927 auto curPaddingLength = binaryChunkLength - bodyBuffer->byteLength;
928 ++GLB_Chunk_count;
929
930 GLB_Chunk binaryChunk;
931 binaryChunk.chunkLength = binaryChunkLength;
932 binaryChunk.chunkType = ChunkType_BIN;
933 AI_SWAP4(binaryChunk.chunkLength);
934
935 size_t bodyOffset = sizeof(GLB_Header) + sizeof(GLB_Chunk) + jsonChunk.chunkLength;
936 outfile->Seek(bodyOffset, aiOrigin_SET);
937 if (outfile->Write(&binaryChunk, 1, sizeof(GLB_Chunk)) != sizeof(GLB_Chunk)) {
938 throw DeadlyExportError("Failed to write body data header!");
939 }
940 if (outfile->Write(bodyBuffer->GetPointer(), 1, bodyBuffer->byteLength) != bodyBuffer->byteLength) {
941 throw DeadlyExportError("Failed to write body data!");
942 }
943 if (curPaddingLength && outfile->Write(&padding, 1, paddingLength) != paddingLength) {
944 throw DeadlyExportError("Failed to write body data padding!");
945 }
946 }
947
948 //
949 // Header
950 //
951
952 GLB_Header header;
953 memcpy(header.magic, AI_GLB_MAGIC_NUMBER, sizeof(header.magic));
954
955 header.version = 2;
956 AI_SWAP4(header.version);
957
958 header.length = uint32_t(sizeof(GLB_Header) + GLB_Chunk_count * sizeof(GLB_Chunk) + jsonChunkLength + binaryChunkLength);
959 AI_SWAP4(header.length);
960
961 outfile->Seek(0, aiOrigin_SET);
962 if (outfile->Write(&header, 1, sizeof(GLB_Header)) != sizeof(GLB_Header)) {
963 throw DeadlyExportError("Failed to write the header!");
964 }
965 }
