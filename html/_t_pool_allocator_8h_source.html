41class TPoolAllocator {
42public:
43 TPoolAllocator();
44 TPoolAllocator(size_t numItems);

50 size_t capacity() const;

58private:
59 struct Pool {
60 size_t m_poolsize;
61 T *m_pool;
62 size_t m_currentIdx;
63 Pool *m_next;
64
65 Pool() :
66 m_poolsize(0u), m_pool(nullptr), m_currentIdx(0u), m_next(nullptr) {
67 // empty
68 }
69
70 Pool(size_t numItems, Pool *prev) :
71 m_poolsize(numItems), m_pool(nullptr), m_currentIdx(0u), m_next(prev) {
72 m_pool = new T[m_poolsize];
73 }
74
75 ~Pool() {
76 delete[] m_pool;
77 m_pool = nullptr;
78 }
79
80 CPPCORE_NONE_COPYING(Pool)
81 };

83 Pool *getFreePool() {
84 Pool *current(m_freeList);
85 if (nullptr != m_freeList) {
86 m_freeList = m_freeList->m_next;
87 }
88 return current;
89 }
90
91 Pool *m_first;
92 Pool *m_current;
93 Pool *m_freeList;
94 size_t m_capacity;
95};

116template <class T>
117inline T *TPoolAllocator<T>::alloc() {
118 if (nullptr == m_current) {
119 return nullptr;
120 }
121
122 if (m_current->m_currentIdx == m_current->m_poolsize) {
123 resize(m_current->m_poolsize);
124 }
125
126 T *ptr(&m_current->m_pool[m_current->m_currentIdx]);
127 m_current->m_currentIdx++;
128
129 return ptr;
130}
131
132template <class T>
133inline void TPoolAllocator<T>::release() {
134 if (nullptr == m_current) {
135 return;
136 }
137
138 Pool *current(m_first);
139 while (nullptr != current) {
140 current->m_currentIdx = 0;
141 current = current->m_next;
142 }
143 m_freeList = m_first->m_next;
144 m_current = m_first;
145}

147template <class T>
148inline void TPoolAllocator<T>::reserve(size_t size) {
149 clear();
150
151 m_first = new Pool(size, nullptr);
152 m_current = m_first;
153
154 m_current->m_pool = new T[size];
155 m_current->m_poolsize = size;
156
157 m_capacity = size;
158}

160template <class T>
161inline void TPoolAllocator<T>::clear() {
162 if (nullptr == m_current) {
163 return;
164 }
165
166 Pool *next(m_first);
167 while (nullptr != next) {
168 Pool *current = next;
169 next = current->m_next;
170 delete current;
171 }
172 m_current = nullptr;
173 m_freeList = nullptr;
174}

203template <class T>
204inline void TPoolAllocator<T>::resize(size_t growSize) {
205 if (nullptr != m_current) {
206 if (growSize < m_current->m_poolsize) {
207 return;
208 }
209 }
210
211 if (nullptr == m_first) {
212 m_first = new Pool(growSize, nullptr);
213 m_current = m_first;
214 m_capacity += m_current->m_poolsize;
215 } else {
216 Pool *pool = getFreePool();
217 if (nullptr == pool) {
218 pool = new Pool(growSize, nullptr);
219 m_capacity += growSize;
220 }
221 m_current->m_next = pool;
222 m_current = m_current->m_next;
223 }
224}
