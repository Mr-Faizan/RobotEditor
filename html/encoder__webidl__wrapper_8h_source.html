82 const draco::Metadata *metadata);
83 bool AddMetadata(draco::PointCloud *pc, const draco::Metadata *metadata);
84
85 private:
86 template <typename DataTypeT>
87 int AddAttribute(draco::PointCloud *pc, draco_GeometryAttribute_Type type,
88 long num_vertices, long num_components,
89 const DataTypeT *att_values,
90 draco::DataType draco_data_type) {
91 if (!pc) {
92 return -1;
93 }
94 std::unique_ptr<draco::PointAttribute> att(new draco::PointAttribute());
95 att->Init(type, num_components, draco_data_type,
96 /* normalized */ false, num_vertices);
97 const int att_id = pc->AddAttribute(std::move(att));
98 draco::PointAttribute *const att_ptr = pc->attribute(att_id);
99
100 for (draco::PointIndex i(0); i < num_vertices; ++i) {
101 att_ptr->SetAttributeValue(att_ptr->mapped_index(i),
102 &att_values[i.value() * num_components]);
103 }
104 if (pc->num_points() == 0) {
105 pc->set_num_points(num_vertices);
106 } else if (pc->num_points() != num_vertices) {
107 return -1;
108 }
109 return att_id;
110 }
111};
